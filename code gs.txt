/* ---------------------- Helpers: Date / Timestamp Formatters ---------------------- */
function formatDateText_(val) {
  if (!val) return "";
  if (typeof val === 'string' && /^\d{2}\/\d{2}\/\d{4}$/.test(val)) {
    const parts = val.split('/');
    const month = parseInt(parts[0]);
    const day = parseInt(parts[1]);
    
    if (month >= 1 && month <= 12 && day >= 1 && day <= 31) {
      return val;
    }
  }
  
  // Date object
  if (val instanceof Date) {
    let m = ("0" + (val.getMonth() + 1)).slice(-2);
    let d = ("0" + val.getDate()).slice(-2);
    let y = val.getFullYear();
    return `${m}/${d}/${y}`;  // mm/dd/yyyy
  }
  
  val = String(val).trim();
  
  // yyyy-mm-dd (ISO format) -> mm/dd/yyyy
  if (/^\d{4}-\d{2}-\d{2}$/.test(val)) {
    let [y, m, d] = val.split("-");
    return `${m}/${d}/${y}`;  // mm/dd/yyyy
  }
  
  // Already in some slash format
  if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(val)) {
    // Check what format it's in
    const parts = val.split('/');
    const first = parseInt(parts[0]);
    const second = parseInt(parts[1]);
    
    // If first > 12, it's dd/mm/yyyy -> convert to mm/dd/yyyy
    if (first > 12 && second <= 12) {
      // dd/mm/yyyy -> mm/dd/yyyy
      return `${parts[1]}/${parts[0]}/${parts[2]}`;
    }
    
    // If second > 12 and first <= 12, it's already mm/dd/yyyy
    if (second > 12 && first <= 12) {
      return val; // Already mm/dd/yyyy
    }
    
    // Otherwise assume it's already mm/dd/yyyy
    return val;
  }
  
  // Try to parse as date
  const dt = new Date(val);
  if (!isNaN(dt)) {
    let m = ("0" + (dt.getMonth() + 1)).slice(-2);
    let d = ("0" + dt.getDate()).slice(-2);
    let y = dt.getFullYear();
    return `${m}/${d}/${y}`;  // mm/dd/yyyy
  }
  
  return val;
}

// Format timestamp for Philippines (12-hour format with AM/PM)
function formatTimestampPH_(d) {
  if (!d) return "";
  
  var date = new Date(d);
  
  // Format: MM/DD/YYYY, HH:MM:SS AM/PM
  return date.toLocaleString('en-PH', {
    month: '2-digit',
    day: '2-digit',
    year: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
    hour12: true
  });
}
// Format timestamp for detailed activity (with day name)
function formatTimestampDetailed_(d) {
  if (!d) return "";
  
  var date = new Date(d);
  
  // Format: Day Month DD YYYY HH:MM:SS AM/PM
  return date.toLocaleString('en-US', {
    weekday: 'short',
    month: 'short',
    day: 'numeric',
    year: 'numeric',
    hour: 'numeric',
    minute: '2-digit',
    second: '2-digit',
    hour12: true
  });
}
/* ---------------------- SESSION MANAGEMENT SYSTEM ---------------------- */
// Store active sessions in PropertiesService for persistence
const SESSIONS = PropertiesService.getScriptProperties();

function generateSessionId() {
  return Utilities.getUuid();
}

function setActiveSession(username, sessionId, deviceInfo) {
  // Get device fingerprint for unique device identification
  let deviceFingerprint = "unknown";
  try {
    if (typeof deviceInfo === 'string') {
      deviceInfo = JSON.parse(deviceInfo);
    }
    // Create a reliable fingerprint using userAgent + screenResolution
    deviceFingerprint = `${deviceInfo?.userAgent || "unknown"}_${deviceInfo?.screenResolution || "unknown"}`;
    console.log(`🔐 Storing device fingerprint for ${username}: ${deviceFingerprint.substring(0, 100)}...`);
  } catch(e) {
    console.error("Error parsing device info for storage:", e);
  }
  
  const sessionData = {
    sessionId: sessionId,
    deviceInfo: deviceInfo || {},
    deviceFingerprint: deviceFingerprint,
    loginTime: new Date().toISOString(),
    lastActivity: new Date().toISOString(),
    ip: "unknown"
  };
  
  SESSIONS.setProperty(`session_${username}`, JSON.stringify(sessionData));
  updateUserSessionInSheet(username, sessionId, "Active");
  
  return sessionData;
}

function getActiveSession(username) {
  const sessionJson = SESSIONS.getProperty(`session_${username}`);
  if (!sessionJson) return null;
  
  try {
    return JSON.parse(sessionJson);
  } catch (e) {
    console.error("Error parsing session:", e);
    return null;
  }
}

function updateSessionActivity(username) {
  const session = getActiveSession(username);
  if (session) {
    session.lastActivity = new Date().toISOString();
    SESSIONS.setProperty(`session_${username}`, JSON.stringify(session));
    
    // Update sheet
    updateUserSessionInSheet(username, session.sessionId, "Active");
  }
}

function invalidateSession(username) {
  // Remove from PropertiesService
  SESSIONS.deleteProperty(`session_${username}`);
  
  // Update sheet to mark as inactive
  updateUserSessionInSheet(username, "", "Inactive");
  
  return true;
}

function updateUserSessionInSheet(username, sessionId, status) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let activitySheet = ss.getSheetByName("UserActivity");
    
    if (!activitySheet) return false;
    
    const dataRange = activitySheet.getDataRange();
    const values = dataRange.getValues();
    const header = values[0];
    
    const usernameCol = header.indexOf("Username");
    const sessionIdCol = header.indexOf("SessionID") !== -1 ? header.indexOf("SessionID") : header.length;
    const statusCol = header.indexOf("Status") !== -1 ? header.indexOf("Status") : header.length + 1;
    
    // Add columns if they don't exist
    if (header.indexOf("SessionID") === -1) {
      activitySheet.getRange(1, header.length + 1).setValue("SessionID");
    }
    if (header.indexOf("Status") === -1) {
      activitySheet.getRange(1, header.length + 2).setValue("Status");
    }
    
    // Find user row
    for (let i = 1; i < values.length; i++) {
      if (String(values[i][usernameCol]).trim() === username) {
        // Update session ID and status
        if (sessionIdCol < activitySheet.getLastColumn()) {
          activitySheet.getRange(i + 1, sessionIdCol + 1).setValue(sessionId);
        }
        if (statusCol < activitySheet.getLastColumn()) {
          activitySheet.getRange(i + 1, statusCol + 1).setValue(status);
        }
        return true;
      }
    }
    
    return false;
  } catch (err) {
    console.error("updateUserSessionInSheet error:", err);
    return false;
  }
}

function checkSessionExpired(session, username) {
  if (!session || !session.lastActivity) return true;
  
  const lastActivity = new Date(session.lastActivity);
  const now = new Date();
  const inactiveMinutes = (now.getTime() - lastActivity.getTime()) / (1000 * 60);
  
  // Session expires after 8 hours of inactivity (keep this)
  const isExpired = inactiveMinutes > 480;
  
  // Log browser close detection (for monitoring)
  if (inactiveMinutes > 3 && inactiveMinutes < 480) {
    console.log(`⚠️ Session for ${username} inactive for ${Math.floor(inactiveMinutes)} minutes - browser likely closed`);
  }
  
  if (isExpired && username) {
    console.log(`Session expired for user ${username} after 8 hours, cleaning up...`);
    invalidateSession(username);
    
    const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
    if (pendingRequests[username]) {
      delete pendingRequests[username];
      SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
    }
  }
  
  return isExpired;
}
function handleHeartbeat(data) {
  try {
    const username = data.username;
    const sessionId = data.sessionId;
    
    if (!username || !sessionId) {
      return { success: false, message: "Missing username or sessionId" };
    }
    
    const session = getActiveSession(username);
    
    if (!session) {
      return { success: false, isLoggedIn: false, message: "No active session" };
    }
    
    if (session.sessionId !== sessionId) {
      return { success: false, isLoggedIn: false, message: "Session ID mismatch" };
    }
    
    // Update last activity
    updateSessionActivity(username);
    
    return { 
      success: true, 
      isLoggedIn: true, 
      message: "Heartbeat received",
      timestamp: new Date().toISOString()
    };
    
  } catch (err) {
    console.error("handleHeartbeat error:", err);
    return { success: false, message: err.toString() };
  }
}

function logUserLoginWithDevice(username, deviceInfo = {}, sessionId) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let activitySheet = ss.getSheetByName("UserActivity");
    
    if (!activitySheet) {
      setupUserActivitySheet();
      activitySheet = ss.getSheetByName("UserActivity");
    }
    
    const now = new Date();
    const deviceString = JSON.stringify(deviceInfo);
    
    // Check if user already has an entry
    const dataRange = activitySheet.getDataRange();
    const values = dataRange.getValues();
    const header = values[0];
    
    const usernameCol = header.indexOf("Username");
    const lastLoginCol = header.indexOf("LastLogin");
    const sessionStartCol = header.indexOf("SessionStart");
    const isOnlineCol = header.indexOf("IsOnline");
    const lastActivityCol = header.indexOf("LastActivity");
    const deviceInfoCol = header.indexOf("DeviceInfo");
    const ipAddressCol = header.indexOf("IPAddress");
    const sessionIdCol = header.indexOf("SessionID");
    const statusCol = header.indexOf("Status");
    
    let userRow = -1;
    for (let i = 1; i < values.length; i++) {
      if (String(values[i][usernameCol]).trim() === username) {
        userRow = i + 1;
        break;
      }
    }
    
    if (userRow > 0) {
      // Update existing user
      activitySheet.getRange(userRow, lastLoginCol + 1).setValue(now);
      activitySheet.getRange(userRow, sessionStartCol + 1).setValue(now);
      activitySheet.getRange(userRow, lastActivityCol + 1).setValue(now);
      activitySheet.getRange(userRow, isOnlineCol + 1).setValue("Online");
      
      if (deviceInfoCol !== -1) {
        activitySheet.getRange(userRow, deviceInfoCol + 1).setValue(deviceString);
      }
      
      if (ipAddressCol !== -1) {
        activitySheet.getRange(userRow, ipAddressCol + 1).setValue("unknown");
      }
      
      if (sessionIdCol !== -1 && sessionId) {
        activitySheet.getRange(userRow, sessionIdCol + 1).setValue(sessionId);
      }
      
      if (statusCol !== -1) {
        activitySheet.getRange(userRow, statusCol + 1).setValue("Active");
      }
    } else {
      // Create new entry
      const newRow = [
        username,           // Username
        now,                // LastLogin
        now,                // LastActivity
        now,                // SessionStart
        "",                 // SessionEnd (empty)
        "Online",           // IsOnline
        deviceString,       // DeviceInfo
        "unknown",          // IPAddress
        sessionId || "",    // SessionID
        "Active"           // Status
      ];
      
      activitySheet.appendRow(newRow);
    }
    
    return true;
  } catch (err) {
    console.error("logUserLoginWithDevice error:", err);
    return false;
  }
}

/* ---------------------- GET USER FULL NAME FROM USERS SHEET ---------------------- */
function getUserFullName(username) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const usersSheet = ss.getSheetByName("Users");
    
    if (!usersSheet) return username;
    
    const dataRange = usersSheet.getDataRange();
    const values = dataRange.getValues();
    const header = values[0];
    
    const usernameIndex = header.findIndex(col => 
      col.toString().trim().toLowerCase() === "username");
    const fullnameIndex = header.findIndex(col => 
      col.toString().trim().toLowerCase() === "fullname" || 
      col.toString().trim().toLowerCase() === "name");
    
    if (usernameIndex !== -1 && fullnameIndex !== -1) {
      for (let i = 1; i < values.length; i++) {
        const row = values[i];
        const rowUsername = String(row[usernameIndex] || "").trim();
        if (rowUsername === username) {
          const fullName = String(row[fullnameIndex] || "").trim();
          return fullName || username;
        }
      }
    }
    
    return username;
  } catch (err) {
    console.error("getUserFullName error:", err);
    return username;
  }
}

/* ---------------------- ENHANCED VALIDATE SESSION WITH FULL NAME ---------------------- */
function validateSessionWithFullName(data) {
  try {
    const username = data.username;
    const sessionId = data.sessionId;
    
    if (!username || !sessionId) {
      return {
        success: false,
        isLoggedIn: false,
        message: "Username and session ID required"
      };
    }
    
    // Use existing session check
    const sessionCheck = checkUserSessionStatus({
      username: username,
      sessionId: sessionId
    });
    
    if (!sessionCheck.success || !sessionCheck.isLoggedIn) {
      return sessionCheck;
    }
    
    // Get user full name
    const fullName = getUserFullName(username);
    
    // Get user role and barangay
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const usersSheet = ss.getSheetByName("Users");
    let role = "Staff";
    let assignedBarangay = "";
    
    if (usersSheet) {
      const dataRange = usersSheet.getDataRange();
      const values = dataRange.getValues();
      const header = values[0];
      
      const usernameIndex = header.findIndex(col => 
        col.toString().trim().toLowerCase() === "username");
      const roleIndex = header.findIndex(col => 
        col.toString().trim().toLowerCase() === "role");
      const barangayIndex = header.findIndex(col => 
        col.toString().trim().toLowerCase() === "assignedbarangay" || 
        col.toString().trim().toLowerCase() === "barangay");
      
      if (usernameIndex !== -1) {
        for (let i = 1; i < values.length; i++) {
          const row = values[i];
          const rowUsername = String(row[usernameIndex] || "").trim();
          if (rowUsername === username) {
            if (roleIndex !== -1) role = String(row[roleIndex] || "").trim() || "Staff";
            if (barangayIndex !== -1) assignedBarangay = String(row[barangayIndex] || "").trim();
            break;
          }
        }
      }
    }
    
    return {
      success: true,
      isLoggedIn: true,
      message: "Session valid",
      user: {
        Username: username,
        FullName: fullName,
        Role: role,
        AssignedBarangay: assignedBarangay
      },
      sessionId: sessionId,
      lastActivity: sessionCheck.lastActivity,
      sessionStart: sessionCheck.sessionStart
    };
    
  } catch (err) {
    console.error("validateSessionWithFullName error:", err);
    return {
      success: false,
      isLoggedIn: false,
      message: "Session validation error: " + err.toString()
    };
  }
}
/* ---------------------- FORCE LOGOUT FUNCTION ---------------------- */
function forceLogoutUser(username, actor = "System", requestId = null) {
  try {
    if (!username) {
      return { success: false, message: "Username is required" };
    }
    
    const session = getActiveSession(username);
    if (!session) {
      return { success: false, message: "No active session found" };
    }
    
    // If this is a confirmation (requestId provided), mark the pending request as confirmed
    if (requestId) {
      const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
      if (pendingRequests[username] && pendingRequests[username].requestId === requestId) {
        pendingRequests[username].status = 'confirmed';
        SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
        console.log(`✅ Marked request ${requestId} as confirmed for user ${username}`);
      }
    }
    
    // Invalidate the session
    invalidateSession(username);
    
    // Update activity sheet
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let activitySheet = ss.getSheetByName("UserActivity");
    if (activitySheet) {
      const dataRange = activitySheet.getDataRange();
      const values = dataRange.getValues();
      const header = values[0];
      const usernameCol = header.indexOf("Username");
      const sessionEndCol = header.indexOf("SessionEnd");
      const isOnlineCol = header.indexOf("IsOnline");
      const statusCol = header.indexOf("Status");
      
      for (let i = 1; i < values.length; i++) {
        if (String(values[i][usernameCol]).trim() === username) {
          activitySheet.getRange(i + 1, sessionEndCol + 1).setValue(new Date());
          activitySheet.getRange(i + 1, isOnlineCol + 1).setValue("Force Logged Out");
          if (statusCol !== -1) {
            activitySheet.getRange(i + 1, statusCol + 1).setValue("Force Logged Out");
          }
          break;
        }
      }
    }
    
    logAuditEvent("Force Logout", actor, "UserActivity", `Forcefully logged out user "${username}"`);
    
    // Only delete the pending request if it was a direct force logout (no requestId)
    if (!requestId) {
      const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
      delete pendingRequests[username];
      SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
    }
    
    return {
      success: true,
      message: "User force logged out from all devices"
    };
    
  } catch (err) {
    console.error("forceLogoutUser error:", err);
    return { success: false, message: "Error: " + err.toString() };
  }
}

// New function: Check if there's a pending force logout request
function checkPendingForceLogout(username) {
  try {
    const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
    const request = pendingRequests[username];
    
    if (request && request.status === 'pending') {
      // Check if request is older than 2 minutes (timeout)
      const requestTime = new Date(request.timestamp);
      const now = new Date();
      const ageSeconds = (now - requestTime) / 1000;
      
      if (ageSeconds > 120) {
        // Request expired
        delete pendingRequests[username];
        SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
        return null;
      }
      
      return request;
    }
    
    return null;
  } catch (err) {
    console.error("checkPendingForceLogout error:", err);
    return null;
  }
}

/* ---------------------- CHECK SESSION STATUS ---------------------- */
function checkUserSessionStatus(data) {
  try {
    const username = data.username;
    const sessionId = data.sessionId;
    
    if (!username) {
      return {
        success: false,
        isLoggedIn: false,
        message: "Username is required"
      };
    }
    
    const session = getActiveSession(username);
    
    if (!session) {
      return {
        success: true,
        isLoggedIn: false,
        message: "No active session found"
      };
    }
    
// Check if session expired
if (checkSessionExpired(session)) {
  // AUTOMATICALLY CLEAR THE EXPIRED SESSION
  invalidateSession(username);
  
  // ALSO CLEAR ANY PENDING FORCE LOGOUT REQUEST FOR THIS USER
  const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
  if (pendingRequests[username]) {
    delete pendingRequests[username];
    SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
    console.log(`Cleared pending force logout request for ${username} due to session expiration`);
  }
  
  logAuditEvent("Session Expired", username, "System", 
               `Session automatically cleared due to inactivity`);
  
  return {
    success: true,
    isLoggedIn: false,
    message: "Session expired due to inactivity and has been cleared",
    expired: true,
    cleared: true
  };
}
    
    // Check if session matches
    if (sessionId && session.sessionId !== sessionId) {
      return {
        success: true,
        isLoggedIn: false,
        message: "Session ID mismatch"
      };
    }
    
    // Update last activity
    updateSessionActivity(username);
    
    return {
      success: true,
      isLoggedIn: true,
      message: "User is currently logged in",
      lastActivity: session.lastActivity,
      sessionStart: session.loginTime,
      sessionId: session.sessionId
    };
    
  } catch (err) {
    console.error("checkUserSessionStatus error:", err);
    return {
      success: false,
      message: "Error checking session status: " + err.toString()
    };
  }
}

function getUserActivity(username = null) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const activitySheet = ss.getSheetByName("UserActivity");
    
    if (!activitySheet) {
      return {
        success: true,
        activities: [],
        message: "No activity data available"
      };
    }
    
    const dataRange = activitySheet.getDataRange();
    const values = dataRange.getValues();
    
    if (values.length < 2) {
      return {
        success: true,
        activities: [],
        message: "No activity records found"
      };
    }
    
    const header = values[0];
    const activities = [];
    
    for (let i = 1; i < values.length; i++) {
      const row = values[i];
      const activity = {};
      
      // Map columns to activity object
      activity.Username = row[header.indexOf("Username")];
      activity.LastLogin = row[header.indexOf("LastLogin")];
      activity.LastActivity = row[header.indexOf("LastActivity")];
      activity.SessionStart = row[header.indexOf("SessionStart")];
      activity.SessionEnd = row[header.indexOf("SessionEnd")];
      activity.IsOnline = row[header.indexOf("IsOnline")];
      activity.DeviceInfo = row[header.indexOf("DeviceInfo")];
      activity.IPAddress = row[header.indexOf("IPAddress")];
      activity.SessionID = row[header.indexOf("SessionID")];
      activity.Status = row[header.indexOf("Status")];
      
      // Calculate time since last activity
      if (activity.LastActivity && activity.LastActivity instanceof Date) {
        const now = new Date();
        const diffMs = now.getTime() - activity.LastActivity.getTime();
        const diffMinutes = Math.floor(diffMs / (1000 * 60));
        const diffHours = Math.floor(diffMinutes / 60);
        
        if (diffMinutes < 1) {
          activity.TimeAgo = "Just now";
        } else if (diffMinutes < 60) {
          activity.TimeAgo = `${diffMinutes} minute(s) ago`;
        } else if (diffHours < 24) {
          activity.TimeAgo = `${diffHours} hour(s) ago`;
        } else {
          const diffDays = Math.floor(diffHours / 24);
          activity.TimeAgo = `${diffDays} day(s) ago`;
        }
      }
      
      // Calculate session duration if session is active
      if (activity.IsOnline === "Online" && activity.SessionStart && 
          activity.SessionStart instanceof Date) {
        const now = new Date();
        const sessionMs = now.getTime() - activity.SessionStart.getTime();
        const sessionMinutes = Math.floor(sessionMs / (1000 * 60));
        const sessionHours = Math.floor(sessionMinutes / 60);
        
        if (sessionMinutes < 60) {
          activity.SessionDuration = `${sessionMinutes} minute(s)`;
        } else {
          activity.SessionDuration = `${sessionHours} hour(s), ${sessionMinutes % 60} minute(s)`;
        }
      } else if (activity.SessionStart && activity.SessionEnd &&
                activity.SessionStart instanceof Date && 
                activity.SessionEnd instanceof Date) {
        const sessionMs = activity.SessionEnd.getTime() - activity.SessionStart.getTime();
        const sessionMinutes = Math.floor(sessionMs / (1000 * 60));
        const sessionHours = Math.floor(sessionMinutes / 60);
        
        if (sessionMinutes < 60) {
          activity.SessionDuration = `${sessionMinutes} minute(s)`;
        } else {
          activity.SessionDuration = `${sessionHours} hour(s), ${sessionMinutes % 60} minute(s)`;
        }
      }
      
      if (username) {
        if (activity.Username === username) {
          activities.push(activity);
        }
      } else {
        activities.push(activity);
      }
    }
    
    // Sort by last activity (newest first)
    activities.sort((a, b) => {
      const dateA = a.LastActivity || a.LastLogin || new Date(0);
      const dateB = b.LastActivity || b.LastLogin || new Date(0);
      return new Date(dateB) - new Date(dateA);
    });
    
    return {
      success: true,
      activities: activities,
      count: activities.length
    };
    
  } catch (err) {
    console.error("getUserActivity error:", err);
    return {
      success: false,
      message: "Error getting user activity: " + err.toString(),
      activities: []
    };
  }
}
/* ---------------------- FILTERED AUDIT LOGS - SHOW ONLY IMPORTANT ACTIONS ---------------------- */
function getAuditLogs() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const logsSheet = ss.getSheetByName("AuditLogs");
    
    if (!logsSheet) {
      return {
        success: true,
        logs: [],
        today: [],
        yesterday: [],
        older: [],
        message: "No logs available"
      };
    }
    
    const dataRange = logsSheet.getDataRange();
    const values = dataRange.getValues();
    
    if (values.length < 2) {
      return {
        success: true,
        logs: [],
        today: [],
        yesterday: [],
        older: [],
        message: "No logs found"
      };
    }
    
    // Extract logs (skip header row)
    const header = values[0];
    const allLogs = [];
    const todayLogs = [];
    const yesterdayLogs = [];
    const olderLogs = [];
    
    // Define IMPORTANT actions that should be shown in Audit Logs
    // These are the actions you want to keep
const IMPORTANT_ACTIONS = [
  "User Login",
  "User Logout", 
  "Create User",
  "Update User",
  "Delete User",
  "Reset Password",
  "Force Logout",
  "Create Record",
  "Update Household",  // For dashboard "Last Updated"
  "Delete Household",  // For dashboard "Last Updated"
  "Append Members",
  "Session Mismatch Logout",
  //"User Activity",     // For user activity tracking
  "Manual Refresh",    // For dashboard refresh button
  "Dashboard Refresh"  // For auto-refresh
];
    
    // Get date ranges
    const now = new Date();
    const today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    const yesterday = new Date(today);
    yesterday.setDate(yesterday.getDate() - 1);
    
    for (let i = 1; i < values.length; i++) {
      const row = values[i];
      const logEntry = {};
      
      // Map row data to object using EXACT header names from spreadsheet
      for (let j = 0; j < header.length; j++) {
        const colName = header[j];
        logEntry[colName] = row[j];
      }
      
      // Also create camelCase versions for backward compatibility
      if (logEntry["Timestamp"]) {
        logEntry.Timestamp = logEntry["Timestamp"];
        logEntry.FormattedTimestamp = formatTimestampPH_(logEntry["Timestamp"]);
      }
      if (logEntry["Action"]) logEntry.Action = logEntry["Action"];
      if (logEntry["Actor"]) logEntry.Actor = logEntry["Actor"];
      if (logEntry["TargetSheet"]) {
        logEntry.TargetSheet = logEntry["TargetSheet"];
        logEntry.Target = logEntry["TargetSheet"];
      }
      if (logEntry["Details"]) logEntry.Details = logEntry["Details"];
      
      // CRITICAL: Only show IMPORTANT actions, filter out "User Activity" noise
      const action = logEntry["Action"] || "";
      const isImportantAction = IMPORTANT_ACTIONS.includes(action);
      
      // Only include logs that are important actions
      if (isImportantAction) {
        // Categorize by date
        if (logEntry["Timestamp"] instanceof Date) {
          const logDate = new Date(logEntry["Timestamp"]);
          const logDay = new Date(logDate.getFullYear(), logDate.getMonth(), logDate.getDate());
          
          if (logDay.getTime() === today.getTime()) {
            logEntry.Category = "Today";
            todayLogs.push(logEntry);
          } else if (logDay.getTime() === yesterday.getTime()) {
            logEntry.Category = "Yesterday";
            yesterdayLogs.push(logEntry);
          } else {
            logEntry.Category = "Older";
            olderLogs.push(logEntry);
          }
        } else {
          logEntry.Category = "Older";
          olderLogs.push(logEntry);
        }
        
        allLogs.push(logEntry);
      }
      // All "User Activity" logs will be filtered out automatically
    }
    
    // Sort by timestamp (newest first) within each category
    const sortByTimestamp = (a, b) => {
      const dateA = a["Timestamp"] || a.Timestamp || new Date(0);
      const dateB = b["Timestamp"] || b.Timestamp || new Date(0);
      return new Date(dateB) - new Date(dateA);
    };
    
    allLogs.sort(sortByTimestamp);
    todayLogs.sort(sortByTimestamp);
    yesterdayLogs.sort(sortByTimestamp);
    olderLogs.sort(sortByTimestamp);
    
    return {
      success: true,
      logs: allLogs,
      today: todayLogs,
      yesterday: yesterdayLogs,
      older: olderLogs,
      count: allLogs.length,
      counts: {
        today: todayLogs.length,
        yesterday: yesterdayLogs.length,
        older: olderLogs.length
      }
    };
  } catch (err) {
    console.error("getAuditLogs error:", err);
    return {
      success: false,
      message: "Error getting logs: " + err.toString(),
      logs: [],
      today: [],
      yesterday: [],
      older: []
    };
  }
}
/* ---------------------- ENHANCED AUDIT LOG FUNCTION ---------------------- */
function logAuditEvent(action, actor, targetSheet, details) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let logsSheet = ss.getSheetByName("AuditLogs");
    
    if (!logsSheet) {
      logsSheet = ss.insertSheet("AuditLogs");
      logsSheet.appendRow(["Timestamp", "Action", "Actor", "TargetSheet", "Details"]);
      
      // Format header
      const headerRange = logsSheet.getRange(1, 1, 1, 5);
      headerRange.setBackground("#0f7a4a");
      headerRange.setFontColor("#ffffff");
      headerRange.setFontWeight("bold");
      
      // Set column widths
      logsSheet.setColumnWidth(1, 180); // Timestamp
      logsSheet.setColumnWidth(2, 150); // Action
      logsSheet.setColumnWidth(3, 150); // Actor (Full Name)
      logsSheet.setColumnWidth(4, 120); // TargetSheet
      logsSheet.setColumnWidth(5, 400); // Details
    }
    
    // Get user's full name from Users sheet
    let actorFullName = actor;
    let actorRole = "Staff";
    
    try {
      const usersSheet = ss.getSheetByName("Users");
      if (usersSheet) {
        const dataRange = usersSheet.getDataRange();
        const values = dataRange.getValues();
        const header = values[0];
        
        const usernameIndex = header.findIndex(col => 
          col.toString().trim().toLowerCase() === "username");
        const fullnameIndex = header.findIndex(col => 
          col.toString().trim().toLowerCase() === "fullname" || 
          col.toString().trim().toLowerCase() === "name");
        const roleIndex = header.findIndex(col => 
          col.toString().trim().toLowerCase() === "role");
        
        if (usernameIndex !== -1) {
          for (let i = 1; i < values.length; i++) {
            const row = values[i];
            const rowUsername = String(row[usernameIndex] || "").trim();
            
            if (rowUsername === actor) {
              if (fullnameIndex !== -1) {
                actorFullName = String(row[fullnameIndex] || "").trim() || actor;
              }
              if (roleIndex !== -1) {
                actorRole = String(row[roleIndex] || "").trim() || "Staff";
              }
              break;
            }
          }
        }
      }
    } catch (e) {
      console.error("Error getting user details for audit log:", e);
    }
    
    const timestamp = new Date();
    
    // Format details
    let formattedDetails = details;
    
    // If details is empty, create meaningful details based on action
    if (!details || details.trim() === "") {
      switch(action) {
        case "User Login":
          formattedDetails = `${actorFullName} (${actorRole}) logged in to the system`;
          break;
        case "User Logout":
          formattedDetails = `${actorFullName} (${actorRole}) logged out from the system`;
          break;
        case "Create User":
          formattedDetails = `${actorFullName} (${actorRole}) created a new user account`;
          break;
        case "Update User":
          formattedDetails = `${actorFullName} (${actorRole}) updated user information`;
          break;
        case "Delete User":
          formattedDetails = `${actorFullName} (${actorRole}) deleted a user account`;
          break;
        case "Reset Password":
          formattedDetails = `${actorFullName} (${actorRole}) reset a user's password`;
          break;
        case "Force Logout":
          formattedDetails = `${actorFullName} (${actorRole}) force logged out a user`;
          break;
        default:
          formattedDetails = `${actorFullName} (${actorRole}) performed: ${action}`;
      }
    }
    
    // Append the log with 5 columns (not 6)
    logsSheet.appendRow([
      timestamp,            // Column 1: Timestamp
      action,               // Column 2: Action
      actorFullName,        // Column 3: Actor (Full Name only)
      targetSheet,          // Column 4: TargetSheet (e.g., "Users", "UserActivity")
      formattedDetails      // Column 5: Details (description of what happened)
    ]);
    
    return true;
  } catch (err) {
    console.error("logAuditEvent error:", err);
    return false;
  }
}

/* ---------------------- SUPER FAST LOGIN FUNCTION ---------------------- */
function handleLogin(param) {
  console.log("=== FAST LOGIN START ===");
  const startTime = Date.now();
  
try {
    
    if (!param.username || !param.pwHash) {
      return { success: false, message: "Missing username or password" };
    }
    
    const username = String(param.username).trim();
    const pwHash = String(param.pwHash).trim();
    
    // ========== STEP 1: GET USER FROM PROPERTIES CACHE (INSTANT) ==========
    let userData = null;
    let rowPassword = null;
    let mustChangePassword = false;
    
    // Try PropertiesService cache first (MUCH FASTER than sheet)
    let userFromProps = getUserFromPropertiesCache(username);
    
    if (userFromProps) {
      console.log("⚡ Properties cache HIT for user:", username);
      rowPassword = userFromProps.passwordHash;
      mustChangePassword = userFromProps.mustChangePassword || false;
      userData = {
        Username: username,
        FullName: userFromProps.fullName,
        Role: userFromProps.role,
        AssignedBarangay: userFromProps.assignedBarangay || ""
      };
    } else {
      // Fall back to ScriptCache (5 min TTL)
      const scriptCache = CacheService.getScriptCache();
      const cacheKey = `user_${username}`;
      const cachedUser = scriptCache.get(cacheKey);
      
      if (cachedUser) {
        console.log("⚡ ScriptCache HIT for user:", username);
        const cached = JSON.parse(cachedUser);
        userData = cached.userData;
        rowPassword = cached.rowPassword;
        mustChangePassword = cached.mustChangePassword;
      } else {
        console.log("📖 Cache MISS, loading from sheet...");
        
        // Load from sheet (this should only happen ONCE per user)
        const ss = SpreadsheetApp.getActiveSpreadsheet();
        const usersSheet = ss.getSheetByName("Users");
        if (!usersSheet) {
          return { success: false, message: "Users sheet not found" };
        }
        
        const lastCol = Math.min(usersSheet.getLastColumn(), 10);
        const dataRange = usersSheet.getRange(1, 1, usersSheet.getLastRow(), lastCol);
        const values = dataRange.getValues();
        
        if (values.length < 2) {
          return { success: false, message: "No users found" };
        }
        
        const headers = values[0];
        
        let usernameIdx = -1, passwordIdx = -1, fullNameIdx = -1, roleIdx = -1, barangayIdx = -1, mustChangeIdx = -1;
        
        for (let i = 0; i < headers.length; i++) {
          const header = String(headers[i]).trim().toLowerCase();
          if (header === "username") usernameIdx = i;
          else if (header === "passwordhash" || header === "password") passwordIdx = i;
          else if (header === "fullname" || header === "name") fullNameIdx = i;
          else if (header === "role") roleIdx = i;
          else if (header === "assignedbarangay" || header === "barangay") barangayIdx = i;
          else if (header === "mustchangepassword") mustChangeIdx = i;
        }
        
        if (usernameIdx === -1 || passwordIdx === -1) {
          return { success: false, message: "Invalid Users sheet structure" };
        }
        
        for (let i = 1; i < values.length; i++) {
          const row = values[i];
          const rowUsername = String(row[usernameIdx] || "").trim();
          
          if (rowUsername === username) {
            rowPassword = String(row[passwordIdx] || "").trim();
            
            if (mustChangeIdx !== -1) {
              const mustChangeValue = row[mustChangeIdx];
              mustChangePassword = (mustChangeValue === true || 
                                   mustChangeValue === "TRUE" || 
                                   String(mustChangeValue).toUpperCase() === "TRUE");
            }
            
            userData = {
              Username: username,
              FullName: (fullNameIdx !== -1 && row[fullNameIdx]) ? String(row[fullNameIdx]).trim() : username,
              Role: (roleIdx !== -1 && row[roleIdx]) ? String(row[roleIdx]).trim().toLowerCase() : "user",
              AssignedBarangay: (barangayIdx !== -1 && row[barangayIdx]) ? String(row[barangayIdx]).trim() : ""
            };
            
            // Store in ScriptCache (5 minutes)
            scriptCache.put(cacheKey, JSON.stringify({
              userData: userData,
              rowPassword: rowPassword,
              mustChangePassword: mustChangePassword
            }), 300);
            
            // Also store in PropertiesCache (persistent)
            updateUserInPropertiesCache(username, {
              passwordHash: rowPassword,
              fullName: userData.FullName,
              role: userData.Role,
              assignedBarangay: userData.AssignedBarangay,
              mustChangePassword: mustChangePassword
            });
            
            break;
          }
        }
        
        if (!userData) {
          return { success: false, message: "User not found" };
        }
      }
    }
    
    // ========== STEP 2: VERIFY PASSWORD ==========
    if (rowPassword !== pwHash) {
      return { success: false, message: "Incorrect password" };
    }
    
    // ========== STEP 3: CHECK SESSION ==========
    const activeSession = getActiveSession(username);
    
    if (activeSession) {
      const lastActivity = new Date(activeSession.lastActivity);
      const inactiveMinutes = (Date.now() - lastActivity.getTime()) / (1000 * 60);
      
      // Get current device fingerprint from the login request
      let currentDeviceFingerprint = "unknown";
      try {
        let deviceInfo = param.deviceInfo;
        if (typeof deviceInfo === 'string') {
          deviceInfo = JSON.parse(deviceInfo);
        }
        // Create a reliable fingerprint using userAgent + screenResolution
        currentDeviceFingerprint = `${deviceInfo?.userAgent || "unknown"}_${deviceInfo?.screenResolution || "unknown"}`;
        console.log(`📱 Current Device Fingerprint: ${currentDeviceFingerprint.substring(0, 100)}...`);
      } catch(e) {
        console.error("Error getting device fingerprint:", e);
      }
      
      // Get stored device fingerprint from active session
      const storedDeviceFingerprint = activeSession.deviceFingerprint || activeSession.deviceId || "";
      console.log(`💾 Stored Device Fingerprint: ${storedDeviceFingerprint.substring(0, 100)}...`);
      
      // Check if this is the SAME device
      const isSameDevice = currentDeviceFingerprint !== "unknown" && 
                           storedDeviceFingerprint !== "" && 
                           currentDeviceFingerprint === storedDeviceFingerprint;
      
      console.log(`🔍 Same device? ${isSameDevice}`);
      console.log(`📊 Inactive minutes: ${inactiveMinutes}`);
      
      // FIRST: Check 8-hour expiry
      if (inactiveMinutes > 480) {
        console.log("✅ Expired session (8 hours), clearing...");
        invalidateSession(username);
      }
      // SECOND: For SAME device - show waiting modal (even within 3 minutes)
      else if (isSameDevice) {
        // For same device, only show waiting modal if session exists within 3 minutes
        if (inactiveMinutes < 3) {
          const waitSeconds = Math.ceil((3 - inactiveMinutes) * 60);
          const remainingMinutesTotal = Math.floor(waitSeconds / 60);
          const remainingSecondsOnly = waitSeconds % 60;
          
          console.log(`✅ SAME DEVICE - Returning SESSION_ACTIVE_NEED_WAIT (wait ${waitSeconds}s)`);
          return {
            success: false,
            message: `Your previous session is still active. Please wait ${remainingMinutesTotal} minute(s) and ${remainingSecondsOnly} second(s) for auto-cleanup.`,
            code: "SESSION_ACTIVE_NEED_WAIT",
            waitSeconds: waitSeconds,
            waitMinutes: remainingMinutesTotal,
            waitSecondsOnly: remainingSecondsOnly
          };
        } else {
          // Same device but inactive for more than 3 minutes - auto-clear
          console.log(`🔄 Same device inactive for ${Math.floor(inactiveMinutes)} minutes - auto-clearing`);
          invalidateSession(username);
        }
      }
      // THIRD: Different device with active session (within 3 minutes) - show force logout
      else if (inactiveMinutes < 3) {
        console.log("❌ DIFFERENT DEVICE - Returning ALREADY_LOGGED_IN");
        const sessionStart = new Date(activeSession.loginTime);
        return {
          success: false,
          message: `Account already logged in on another device. Session started at ${formatTimestampPH_(sessionStart)}.`,
          code: "ALREADY_LOGGED_IN",
          sessionStarted: formatTimestampPH_(sessionStart),
          lastActivity: formatTimestampPH_(new Date(activeSession.lastActivity)),
          inactiveMinutes: Math.floor(inactiveMinutes)
        };
      }
      // FOURTH: Auto-clear inactive session (over 3 minutes) - regardless of device
      else {
        console.log(`🔄 Session inactive for ${Math.floor(inactiveMinutes)} minutes - auto-clearing`);
        invalidateSession(username);
      }
    }
    
    // ========== STEP 4: CREATE NEW SESSION ==========
    const sessionId = generateSessionId();
    
    let deviceInfo = {};
    try {
      if (param.deviceInfo) {
        deviceInfo = typeof param.deviceInfo === 'string' ? 
                    JSON.parse(param.deviceInfo) : param.deviceInfo;
      }
    } catch (e) {
      console.error("Error parsing device info:", e);
    }
    
    setActiveSession(username, sessionId, deviceInfo);
    
    // FIRE AND FORGET - Don't wait for these to complete
    Promise.resolve().then(() => {
      try {
        logUserLoginWithDevice(username, deviceInfo, sessionId);
        logAuditEvent("User Login", username, "System", "User logged in");
      } catch(e) { console.error("Background logging error:", e); }
    });
    
    // Clear any pending force logout requests
    const pendingCleanup = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
    if (pendingCleanup[username]) {
      delete pendingCleanup[username];
      SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingCleanup));
    }
    
    const elapsed = Date.now() - startTime;
    console.log(`✅ Login successful in ${elapsed}ms for: ${username}`);
    
    return {
      success: true,
      message: "Login successful",
      user: userData,
      sessionId: sessionId,
      mustChangePassword: mustChangePassword,
      timeMs: elapsed
    };
    
  } catch (err) {
    console.error("Login error:", err.toString());
    return { 
      success: false, 
      message: "Server error: " + err.toString() 
    };
  }
}

function checkForceLogoutStatus(data) {
  try {
    const username = data.username;
    const requestId = data.requestId;
    
    const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
    const request = pendingRequests[username];
    
    if (!request || request.requestId !== requestId) {
      return { confirmed: false, cancelled: true };
    }
    
    if (request.status === 'confirmed') {
      return { confirmed: true, cancelled: false };
    } else if (request.status === 'pending') {
      return { confirmed: false, cancelled: false, waiting: true };
    } else {
      return { confirmed: false, cancelled: true };
    }
  } catch (err) {
    return { confirmed: false, cancelled: true };
  }
}

    if (values.length < 2) return false;
    
    const headers = values[0];
    
    let usernameIdx = -1, passwordIdx = -1, fullNameIdx = -1, roleIdx = -1, barangayIdx = -1, mustChangeIdx = -1;
    
    for (let i = 0; i < headers.length; i++) {
      const header = String(headers[i]).trim().toLowerCase();
      if (header === "username") usernameIdx = i;
      else if (header === "passwordhash" || header === "password") passwordIdx = i;
      else if (header === "fullname" || header === "name") fullNameIdx = i;
      else if (header === "role") roleIdx = i;
      else if (header === "assignedbarangay" || header === "barangay") barangayIdx = i;
      else if (header === "mustchangepassword") mustChangeIdx = i;
    }
    
    let loadedCount = 0;
    const props = PropertiesService.getScriptProperties();
    
    for (let i = 1; i < values.length; i++) {
      const row = values[i];
      const username = String(row[usernameIdx] || "").trim();
      if (!username) continue;
      
      const mustChangePassword = (mustChangeIdx !== -1 && (row[mustChangeIdx] === true || row[mustChangeIdx] === "TRUE"));
      
      const userData = {
        passwordHash: String(row[passwordIdx] || "").trim(),
        fullName: (fullNameIdx !== -1 && row[fullNameIdx]) ? String(row[fullNameIdx]).trim() : username,
        role: (roleIdx !== -1 && row[roleIdx]) ? String(row[roleIdx]).trim().toLowerCase() : "user",
        assignedBarangay: (barangayIdx !== -1 && row[barangayIdx]) ? String(row[barangayIdx]).trim() : "",
        mustChangePassword: mustChangePassword
      };
      
      props.setProperty(`user_props_${username}`, JSON.stringify(userData));
      loadedCount++;
    }
    
    props.setProperty('users_cache_timestamp', Date.now().toString());
    props.setProperty('users_cache_version', '2');
    
    const elapsed = Date.now() - startTime;
    console.log(`✅ Loaded ${loadedCount} users to Properties cache in ${elapsed}ms`);
    return true;
    
  } catch (err) {
    console.error("Error loading users to cache:", err);
    return false;
  }
}

// Cache clearing function for user data (clears BOTH caches)
function clearUserCache(username) {
  // Clear ScriptCache
  const cache = CacheService.getScriptCache();
  cache.remove(`user_${username}`);
  
  // Clear PropertiesCache
  removeUserFromPropertiesCache(username);
  
  console.log(`Cleared all caches for user: ${username}`);
}

function getUsers() {
  try {
    const props = PropertiesService.getScriptProperties();
    const allProps = props.getProperties();
    const users = [];
    let adminCount = 0;
    let staffCount = 0;
    
    // Get only user properties (start with 'user_props_')
    for (const key in allProps) {
      if (key.startsWith('user_props_') && 
          key !== 'user_props_cache_timestamp' && 
          key !== 'user_props_cache_version') {
        try {
          const userData = JSON.parse(allProps[key]);
          const username = key.substring(11); // Remove 'user_props_' prefix (11 chars)
          
          users.push({
            Username: username,
            FullName: userData.fullName,
            Role: userData.role,
            AssignedBarangay: userData.assignedBarangay || ""
          });
          
          if (userData.role === "Admin" || userData.role === "admin") {
            adminCount++;
          } else {
            staffCount++;
          }
        } catch(e) {
          console.error("Error parsing user:", key, e);
        }
      }
    }
    
    console.log(`✅ Loaded ${users.length} users from cache (${adminCount} Admins, ${staffCount} Staff)`);
    
    return {
      success: true,
      users: users,
      count: users.length,
      totalCount: users.length,
      adminCount: adminCount,
      staffCount: staffCount
    };
    
  } catch (err) {
    console.error("getUsers error:", err.toString());
    return {
      success: false,
      message: "Error getting users: " + err.toString(),
      users: [],
      totalCount: 0,
      adminCount: 0,
      staffCount: 0
    };
  }
}

function changePassword(data) {
  try {
    const username = data.username;
    const currentPasswordHash = data.currentPasswordHash;
    const newPasswordHash = data.newPasswordHash;
    const sessionId = data.sessionId;
    const actor = data.actor || username;
    
    if (!username || !newPasswordHash) {
      return {
        success: false,
        message: "Username and new password are required"
      };
    }
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const usersSheet = ss.getSheetByName("Users");
    
    if (!usersSheet) {
      return {
        success: false,
        message: "Users sheet not found"
      };
    }
    
    const dataRange = usersSheet.getDataRange();
    const values = dataRange.getValues();
    const header = values[0];
    
    const col = {};
    header.forEach((h, i) => {
      if (h && typeof h === 'string') {
        col[h.trim().toLowerCase()] = i;
      }
    });
    
    const usernameCol = col.username;
    const passwordCol = col.passwordhash || col.password;
    const mustChangeCol = col.mustchangepassword;
    const fullNameCol = col.fullname || col.name;
    const roleCol = col.role;
    const barangayCol = col.assignedbarangay || col.barangay;

      const row = values[i];
      const rowUsername = String(row[usernameCol] || "").trim();
      
      if (rowUsername === username) {
        if (currentPasswordHash) {
          const currentStoredHash = String(row[passwordCol] || "").trim();
          if (currentStoredHash !== currentPasswordHash) {
            return {
              success: false,
              message: "Current password is incorrect"
            };
          }
        }
        
        row[passwordCol] = newPasswordHash;
        
        if (mustChangeCol !== undefined) {
          row[mustChangeCol] = "FALSE";
        }
        
        usersSheet.getRange(i + 1, 1, 1, row.length).setValues([row]);
        
        const oldSession = getActiveSession(username);
        if (oldSession) {
          invalidateSession(username);
        }
        
        const userData = {
          Username: username,
          FullName: (fullNameCol !== undefined && row[fullNameCol]) ? 
                   String(row[fullNameCol]).trim() : username,

                String(row[roleCol]).trim().toLowerCase() : "user",

        const newSessionId = generateSessionId();

        logAuditEvent("Password Changed", actor, "Security", 
                     `User "${username}" changed password. All existing sessions were terminated.`);

        const cached = getUserFromPropertiesCache(username);
        if (cached) {
          cached.passwordHash = newPasswordHash;
          cached.mustChangePassword = false;
          updateUserInPropertiesCache(username, cached);
        }
        
        clearUserCache(username);
        
        return {
          success: true,
          message: "Password changed successfully",
          user: userData,
          sessionId: newSessionId,
          sessionsTerminated: true
        };
      }
    }
    
    return {
      success: false,
      message: "User not found"
    };
    
  } catch (err) {
    console.error("changePassword error:", err);
    return {
      success: false,
      message: "Error changing password: " + err.toString()
    };
  }
}

function handleLogout(data) {

        console.log(`✅ Session invalidated for ${username}`);
      }
    }
    
    return { success: true, message: "Logged out successfully" };
    
  } catch (err) {
    console.error("handleLogout error:", err);

function doGet(e) {
  // Handle preflight OPTIONS request
  if (e && e.method === 'options') {
    return doOptions(e);
  }
  
  try {
    console.log("=== doGet START ===");
    console.log("Parameters:", JSON.stringify(e ? e.parameter : {}));
    
    var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : "";
    var callback = (e && e.parameter && e.parameter.callback) ? e.parameter.callback : "";
    
    console.log("Action:", action, "Callback:", callback);
    
    var result;
    
    // Handle different actions
    switch(action) {
      case "getRecords":
        result = getRecords(e.parameter);
        break;
        
      case "test":
        result = { 
          success: true, 
          message: "API is working!",
          timestamp: new Date().toISOString()
        };
        break;
        
      case "getHouseholdsVirtual":
        result = getHouseholdsVirtual(e.parameter);
        break;
      
      case "getHouseholdCount":
        result = getHouseholdCount(e.parameter);
        break;

      case "getHouseholdsByHeadPaginated":
        result = getHouseholdsByHeadPaginated(e.parameter);
        break;

      case "login":
        var loginData = {
          username: e.parameter.username,
          pwHash: e.parameter.pwHash,
          deviceInfo: e.parameter.deviceInfo ? JSON.parse(e.parameter.deviceInfo) : {}
        };
        result = handleLogin(loginData);
        break;

      case "changePassword":
        var changeData = {
          username: e.parameter.username,
          currentPasswordHash: e.parameter.currentPasswordHash,
          newPasswordHash: e.parameter.newPasswordHash
        };
        result = changePassword(changeData);
        break;
    
      case "checkSessionStatus":
        result = checkUserSessionStatus({
          username: e.parameter.username,
          sessionId: e.parameter.sessionId
        });
        break;
    
      case "validateSession":
        result = checkUserSessionStatus({
          username: e.parameter.username,
          sessionId: e.parameter.sessionId
        });
        break;
    
      case "logActivity":
        logAuditEvent("User Activity", e.parameter.actor || e.parameter.username, "UserActivity", e.parameter.action || "Activity");
        logUserActivity(e.parameter.username);
        result = { success: true, message: "Activity logged" };
        break;
    
      case "forceLogout":
        result = {
          success: forceLogoutUser(e.parameter.username, e.parameter.actor || "System"),
          message: "User force logged out from all devices"
        };
        break;

      case "chartData":
        result = getAccurateChartData();
        break;
        
      case "users":
        result = getUsers();
        break;
        
      case "logs":
        result = getAuditLogs();
        break;
        
      case "dashboardLogs":
        result = getDashboardLogs();
        break;
        
      case "userActivity":
        var username = e.parameter.username || null;
        result = getUserActivity(username);
        break;
        
      case "sessionCheck":
        result = sessionCheck();
        break;
        
      default:
        result = { 
          success: true, 
          message: "API Online - Available actions: getRecords, users, logs, dashboardLogs, userActivity, test, sessionCheck",
          timestamp: new Date().toISOString()
        };
    }
    
    console.log("Result:", JSON.stringify(result).substring(0, 200) + "...");
    
    // Handle JSONP callback (NO setHeader here - JSONP bypasses CORS)
    if (callback) {
      console.log("Returning JSONP response");
      var jsonpResponse = callback + '(' + JSON.stringify(result) + ')';
      var jsonpOutput = ContentService.createTextOutput(jsonpResponse);
      jsonpOutput.setMimeType(ContentService.MimeType.JAVASCRIPT);
      return jsonpOutput;
    } else {
      // Use sendJson for CORS headers (sendJson already has setHeader)
      console.log("Returning JSON response via sendJson");
      return sendJson(result);
    }
    
  } catch (err) {
    console.error("doGet ERROR:", err.toString(), err.stack);
    
    var errorResult = { 
      success: false, 
      message: "Server error: " + err.toString(),
      timestamp: new Date().toISOString()
    };
    
    if (e && e.parameter && e.parameter.callback) {
      // JSONP response - no CORS needed
      var errJsonpOutput = ContentService.createTextOutput(e.parameter.callback + '(' + JSON.stringify(errorResult) + ')');
      errJsonpOutput.setMimeType(ContentService.MimeType.JAVASCRIPT);
      return errJsonpOutput;
    } else {
      // Use sendJson for CORS headers
      return sendJson(errorResult);
    }
  }
}
// ============ EDIT LOCK SYSTEM ============
const EDIT_LOCKS = PropertiesService.getScriptProperties();

// Lock a household when user starts editing
function lockHouseholdForEdit(data) {
  try {
    const dataID = data.dataID;
    const username = data.username;
    const fullName = data.fullName || username;
    const sessionId = data.sessionId;
    
    if (!dataID || !username) {
      return { success: false, message: "Missing dataID or username" };
    }
    
    const lockKey = `edit_lock_${dataID}`;
    const existingLock = EDIT_LOCKS.getProperty(lockKey);
    
    if (existingLock) {
      const lockData = JSON.parse(existingLock);
      const lockAge = (Date.now() - lockData.timestamp) / 1000;
      
      // Lock expires after 3 minutes (reduced from 5 for faster cleanup)
      if (lockAge < 180) {
        // Same user? Refresh lock
        if (lockData.username === username && lockData.sessionId === sessionId) {
          lockData.timestamp = Date.now();
          EDIT_LOCKS.setProperty(lockKey, JSON.stringify(lockData));
          return { success: true, isLocked: false, message: "Lock refreshed" };
        }
        
        // Different user - locked
        return { 
          success: false, 
          isLocked: true, 
          lockedBy: lockData.fullName || lockData.username,
          lockedAt: lockData.timestamp,
          message: `Being edited by ${lockData.fullName || lockData.username}`
        };
      }
      // Lock expired, delete it
      EDIT_LOCKS.deleteProperty(lockKey);
    }
    
    // Create new lock
    const lockData = {
      username: username,
      fullName: fullName,
      sessionId: sessionId,
      timestamp: Date.now(),
      dataID: dataID
    };
    
    EDIT_LOCKS.setProperty(lockKey, JSON.stringify(lockData));
    return { success: true, isLocked: false, message: "Household locked" };
    
  } catch (err) {
    console.error("lockHouseholdForEdit error:", err);
    return { success: false, message: err.toString() };
  }
}

// Release lock when user finishes editing
function unlockHousehold(data) {
  try {
    const dataID = data.dataID;
    const username = data.username;
    const sessionId = data.sessionId;
    
    if (!dataID) {
      return { success: false, message: "Missing dataID" };
    }
    
    const lockKey = `edit_lock_${dataID}`;
    const existingLock = EDIT_LOCKS.getProperty(lockKey);
    
    if (existingLock) {
      const lockData = JSON.parse(existingLock);
      // Only release if it's the same user who locked it
      if (lockData.username === username && lockData.sessionId === sessionId) {
        EDIT_LOCKS.deleteProperty(lockKey);
        console.log(`🔓 Lock released for household ${dataID} by ${username}`);
      }
    }
    
    return { success: true, message: "Lock released" };
    
  } catch (err) {
    console.error("unlockHousehold error:", err);
    return { success: false, message: err.toString() };
  }
}

// Check lock status (for polling)
function checkEditLockStatus(data) {
  try {
    const dataID = data.dataID;
    const currentUser = data.username;
    
    if (!dataID) {
      return { success: false, message: "Missing dataID" };
    }
    
    const lockKey = `edit_lock_${dataID}`;
    const existingLock = EDIT_LOCKS.getProperty(lockKey);
    
    if (!existingLock) {
      return { success: true, isLocked: false };
    }
    
    const lockData = JSON.parse(existingLock);
    const lockAge = (Date.now() - lockData.timestamp) / 1000;
    
    // Auto-expire locks after 5 minutes (in case user closes browser)
    if (lockAge > 300) {
      EDIT_LOCKS.deleteProperty(lockKey);
      return { success: true, isLocked: false };
    }
    
    // Check if current user is the locker
    const isCurrentUser = lockData.username === currentUser;
    
    return {
      success: true,
      isLocked: true,
      lockedBy: lockData.fullName || lockData.username,
      lockedAt: lockData.timestamp,
      isCurrentUser: isCurrentUser
    };
    
  } catch (err) {
    console.error("checkEditLockStatus error:", err);
    return { success: false, message: err.toString() };
  }
}

// Clean up stale locks (run via time trigger every hour)
function cleanupStaleEditLocks() {
  const allProps = EDIT_LOCKS.getProperties();
  let cleaned = 0;
  const now = Date.now();
  const LOCK_TIMEOUT = 300000; // 5 minutes in milliseconds
  
  for (const key in allProps) {
    if (key.startsWith('edit_lock_')) {
      try {
        const lockData = JSON.parse(allProps[key]);
        if (now - lockData.timestamp > LOCK_TIMEOUT) {
          EDIT_LOCKS.deleteProperty(key);
          cleaned++;
          console.log(`🧹 Cleaned stale lock for household ${lockData.dataID}`);
        }
      } catch(e) {
        EDIT_LOCKS.deleteProperty(key);
        cleaned++;
      }
    }
  }
  
  console.log(`✅ Cleaned ${cleaned} stale edit locks`);
  return { success: true, cleaned: cleaned };
}
function doPost(e) {
  // Handle preflight OPTIONS request
  if (e && e.method === 'options') {
    return doOptions(e);
  }
  
  try {
    var data = {};

    // handle JSON body
    if (e.postData && e.postData.type && e.postData.type.indexOf("application/json") !== -1) {
      try { data = JSON.parse(e.postData.contents || "{}"); } catch (err) { data = {}; }
    } else if (e.parameter) {
      for (var k in e.parameter) data[k] = e.parameter[k];
    } else if (e.postData && e.postData.type && e.postData.type.indexOf("multipart/form-data") !== -1) {
      // Parse multipart form data for file uploads
      const boundary = e.postData.type.split("boundary=")[1];
      if (boundary) {
        const rawData = e.postData.contents;
        const parts = rawData.split("--" + boundary);
        
        for (let part of parts) {
          if (part.includes("Content-Disposition: form-data")) {
            const nameMatch = part.match(/name="([^"]+)"/);
            if (nameMatch) {
              const fieldName = nameMatch[1];
              
              if (part.includes("filename=")) {
                const filenameMatch = part.match(/filename="([^"]+)"/);
                const fileName = filenameMatch ? filenameMatch[1] : "";
                const headersEnd = part.indexOf("\r\n\r\n");
                
                if (headersEnd !== -1) {
                  let fileData = part.substring(headersEnd + 4);
                  fileData = fileData.replace(/\r\n$/, '');
                  data[fieldName] = fileData;
                  data[fieldName + "_name"] = fileName;
                  data[fieldName + "_type"] = "file";
                }
              } else {
                const valueMatch = part.match(/\r\n\r\n(.*?)(?:\r\n--|$)/s);
                if (valueMatch) {
                  data[fieldName] = valueMatch[1];
                }
              }
            }
          }
        }
      }
    }

    // If head / members are strings containing JSON, parse them
    if (typeof data.head === "string") {
      try { data.head = JSON.parse(data.head); } catch (err) {}
    }
    if (typeof data.members === "string") {
      try { data.members = JSON.parse(data.members); } catch (err) { data.members = []; }
    }

    console.log("=== doPost: Received data ===");
    console.log("Action:", data.action);

    var action = (data.action || "").toString();

      case "checkEditLockStatus":

      // ============ END NEW CASES ============

      case "getHouseholdsByHeadPaginated":

      case "uploadCalamityFile":
        console.log("📡 Processing uploadCalamityFile request");
        const uploadResult = uploadCalamityFileFromForm(data);
        console.log("📡 Upload result:", uploadResult);
        return sendJson(uploadResult);

      case "createRecord":

// Function to check if there's a pending force logout request
function checkForceLogoutRequest(data) {
  console.log("=== checkForceLogoutRequest CALLED ===");
  console.log("Username:", data.username);
  
  try {
    const username = data.username;
    
    if (!username) {
      return { success: false, message: "Username is required" };
    }
    
    const pendingRequests = JSON.parse(SESSIONS.getProperty('pendingForceLogoutRequests') || '{}');
    console.log("Current pending requests:", JSON.stringify(pendingRequests));
    
    const request = pendingRequests[username];
    
    if (!request || request.status !== 'pending') {
      console.log("No pending request found for:", username);
      return { success: true, hasRequest: false };
    }
    
    // Check if request is older than 2 minutes (timeout)
    const requestTime = new Date(request.timestamp);
    const now = new Date();
    const ageSeconds = (now - requestTime) / 1000;
    
    console.log("Request age in seconds:", ageSeconds);
    
    if (ageSeconds > 120) {
      console.log("Request expired, deleting...");
      delete pendingRequests[username];
      SESSIONS.setProperty('pendingForceLogoutRequests', JSON.stringify(pendingRequests));
      return { success: true, hasRequest: false };
    }
    
    console.log("✅ Returning pending request for:", username);
    return {
      success: true,
      hasRequest: true,
      requestId: request.requestId,
      requestedBy: request.requestedBy,
      timestamp: request.timestamp
    };
  } catch (err) {
    console.error("checkForceLogoutRequest error:", err);
    return { success: false, message: err.toString() };
  }
}

// ============ SHARED CACHE USING SCRIPT PROPERTIES (WORKS FOR ALL USERS) ============
function getNextDataIDFast(master) {
  const props = PropertiesService.getScriptProperties();
  const cacheKey = 'id_cache';
  
  // Get current shared cache
  let cacheJson = props.getProperty(cacheKey);
  let idCache = cacheJson ? JSON.parse(cacheJson) : [];
  
  // If shared cache has IDs, use one (THIS IS THE FAST PATH - 1ms)
  if (idCache && idCache.length > 0) {
    const nextId = idCache.shift();
    props.setProperty(cacheKey, JSON.stringify(idCache));
    console.log(`⚡ Shared cache ID: ${nextId} (${idCache.length} left)`);
    return Number(nextId);
  }
  
  // Refill shared cache (only ONE user will do this thanks to lock)
  console.log("📦 Refilling shared ID cache...");
  
  let counterSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("__ID_Gen__");
  
  if (!counterSheet) {
    counterSheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet("__ID_Gen__");
    counterSheet.getRange(1, 1).setValue("LastID");
    counterSheet.getRange(1, 2).setValue(0);
    counterSheet.hideSheet();
    SpreadsheetApp.flush();
  }
  
  // Use LockService to ensure only ONE user refills the cache
  const lock = LockService.getScriptLock();
  let lockAcquired = false;
  
  try {
    lockAcquired = lock.waitLock(3000); // Wait up to 3 seconds
    
    if (!lockAcquired) {
      // Fallback: get single ID from counter sheet
      let currentMax = counterSheet.getRange(1, 2).getValue();
      let nextID = Number(currentMax) + 1;
      counterSheet.getRange(1, 2).setValue(nextID);
      SpreadsheetApp.flush();
      console.log(`⚠️ Lock timeout - direct ID: ${nextID}`);
      return nextID;
    }
    
    // Double-check cache after acquiring lock (another user might have filled it)
    cacheJson = props.getProperty(cacheKey);
    idCache = cacheJson ? JSON.parse(cacheJson) : [];
    
    if (idCache && idCache.length > 0) {
      const nextId = idCache.shift();
      props.setProperty(cacheKey, JSON.stringify(idCache));
      console.log(`⚡ Cache filled by another user, got: ${nextId}`);
      return Number(nextId);
    }
    
    // Get current counter value
    let currentMax = counterSheet.getRange(1, 2).getValue();
    let batchStart = Number(currentMax) + 1;
    let batchSize = 250; // Generate 100 IDs at once
    
    // Update counter sheet
    counterSheet.getRange(1, 2).setValue(batchStart + batchSize - 1);
    SpreadsheetApp.flush();
    
    // Create shared cache
    idCache = [];
    for (let i = 0; i < batchSize; i++) {
      idCache.push(batchStart + i);
    }
    
    // Store in ScriptProperties (shared across ALL users)
    props.setProperty(cacheKey, JSON.stringify(idCache));
    
    const nextId = idCache.shift();
    props.setProperty(cacheKey, JSON.stringify(idCache));
    
    console.log(`✅ Shared batch loaded: ${batchStart} to ${batchStart + batchSize - 1}`);
    console.log(`📤 Returning ID: ${nextId}`);
    
    return Number(nextId);
    
  } catch (e) {
    console.error("Error in ID generation:", e);
    // Ultimate fallback - use timestamp + random (guaranteed unique)
    const fallbackId = parseInt(Date.now().toString().slice(-6) + Math.floor(Math.random() * 1000).toString().padStart(3, '0'));
    console.log(`⚠️ Emergency fallback ID: ${fallbackId}`);
    return fallbackId;
    
  } finally {
    if (lockAcquired) {
      lock.releaseLock();
    }
  }
}
function resetAndInitCounter() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const master = ss.getSheetByName("Master");
  const counterSheet = ss.getSheetByName("__ID_Gen__");
  const props = PropertiesService.getScriptProperties();
  
  // Find max ID
  const lastRow = master.getLastRow();
  let maxID = 0;
  
  if (lastRow > 1) {
    // Scan all rows to find true max
    const ids = master.getRange(2, 1, lastRow - 1, 1).getValues();
    for (let i = 0; i < ids.length; i++) {
      const id = Number(ids[i][0]);
      if (!isNaN(id) && id > maxID) maxID = id;
    }
  }
  
  console.log(`Current max ID: ${maxID}`);
  
  // Set counter to maxID (next will be maxID + 1)
  if (!counterSheet) {
    const newSheet = ss.insertSheet("__ID_Gen__");
    newSheet.getRange(1, 1).setValue("LastID");
    newSheet.getRange(1, 2).setValue(maxID);
    newSheet.hideSheet();
  } else {
    counterSheet.getRange(1, 2).setValue(maxID);
  }
  
  // Clear shared cache
  props.deleteProperty('id_cache');
  props.deleteProperty('id_counter');
  
  console.log(`✅ Counter set to: ${maxID}`);
  console.log(`✅ Next ID will be: ${maxID + 1}`);
  console.log(`✅ Shared cache cleared`);
  
  return { maxID, nextID: maxID + 1 };
}

      if (colIndex["Transportation"] !== undefined) memberRow[colIndex["Transportation"]] = member["Transportation"] || "";
      if (colIndex["Submitted By"] !== undefined) memberRow[colIndex["Submitted By"]] = submittedByFullName;
      if (colIndex["Submitted On"] !== undefined) memberRow[colIndex["Submitted On"]] = timestamp;
      
      rows.push(memberRow);
    }

    // Write ALL rows in ONE operation
    if (rows.length > 0) {
      master.getRange(master.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows);

          bs.getRange(1, 1, 1, header.length).setValues([header]);
          bs.getRange(2, 1, rows.length, rows[0].length).setValues(rows);
        }
      }
    }

                 `Appended ${rows.length} members to Data ID ${dataID}`);

  } catch (err) {
    console.error("appendMembers error:", err);
    return { success: false, message: err.toString() };
  }

/* ---------------------- listHouseholds ---------------------- */
function listHouseholds() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var master = ss.getSheetByName("Master");
    if (!master) return { success: false, message: "Master sheet not found" };

    var header = master.getRange(1, 1, 1, master.getLastColumn()).getValues()[0];
    var rowCount = master.getLastRow() - 1;
    var values = rowCount > 0 ? master.getRange(2, 1, rowCount, header.length).getValues() : [];

    var groups = {};
    for (var i = 0; i < values.length; i++) {
      var r = values[i];
      var rowNum = i + 2;
      var obj = {};
      for (var j = 0; j < header.length; j++) {
        if (header[j] === "Birthdate") {
          obj[header[j]] = formatDateText_(r[j]);
        } else if (header[j] === "Submitted On") {
          obj[header[j]] = formatTimestampPH_(r[j]);
        } else if (header[j].endsWith("_Date") && r[j]) {
          // Format government assistance date fields
          obj[header[j]] = formatDateText_(r[j]);
        } else {
          obj[header[j]] = r[j];
        }
      }

      var dataID = String(obj["Data ID"] || "");
      if (!dataID) continue;
      var isHead = (!obj["Name of Household Member/s"] || String(obj["Name of Household Member/s"]).trim() === "");

      if (!groups[dataID]) groups[dataID] = { dataID: dataID, head: null, headRowNumber: null, members: [] };

      if (isHead && !groups[dataID].head) {
        groups[dataID].head = obj;
        groups[dataID].headRowNumber = rowNum;
      } else {
        obj.__rowNumber = rowNum;
        groups[dataID].members.push(obj);
      }
    }

    var out = [];
    for (var k in groups) {
      var g = groups[k];
      if (!g.head && g.members.length > 0) {
        g.head = g.members.shift();
        g.headRowNumber = g.head.__rowNumber;
        delete g.head.__rowNumber;
      } else if (g.head && g.head.__rowNumber) {
        delete g.head.__rowNumber;
      }
      out.push(g);
    }

    return { success: true, households: out, header: header };

  } catch (err) {
    return { success: false, message: err.toString() };
  }
}

/* ---------------------- deleteHousehold ---------------------- */
function deleteHousehold(dataID, actor = "System") {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var master = ss.getSheetByName("Master");
    if (!master) return { success: false, message: "Master sheet not found" };

    var header = master.getRange(1, 1, 1, master.getLastColumn()).getValues()[0];
    var values = master.getRange(2, 1, master.getLastRow() - 1, header.length).getValues();

    var barangayFound = null;
    var householdHead = "";
    var delMaster = 0;

    // First pass: collect info
    for (var i = values.length - 1; i >= 0; i--) {
      if (String(values[i][0]) === String(dataID)) {
        if (!barangayFound) {
          barangayFound = values[i][header.indexOf("Barangay")];
          householdHead = values[i][header.indexOf("Name of Household Head")] || "";
        }
        delMaster++;
      }
    }

    // Delete master rows bottom-up
    for (var i = values.length - 1; i >= 0; i--) {
      if (String(values[i][0]) === String(dataID)) {
        master.deleteRow(i + 2);
      }
    }

    var delBarangay = 0;
    if (barangayFound) {
      var bs = ss.getSheetByName(String(barangayFound));
      if (bs) {
        var bh = bs.getRange(1, 1, 1, bs.getLastColumn()).getValues()[0];
        var br = bs.getRange(2, 1, bs.getLastRow() - 1, bh.length).getValues();
        for (var j = br.length - 1; j >= 0; j--) {
          if (String(br[j][0]) === String(dataID)) {
            bs.deleteRow(j + 2);
            delBarangay++;
          }
        }
      }
    }

    // Log the deletion WITH BARANGAY NAME - UPDATED LINE
    logAuditEvent("Delete Household", actor, String(barangayFound) || "Unknown", 
                 `Deleted Data ID ${dataID} for ${householdHead || "Unknown"} in ${String(barangayFound) || "Unknown Barangay"}`);

    invalidateAllCaches();

    return { 
      success: true, 
      message: "Deleted " + delMaster + " rows in Master and " + delBarangay + " in Barangay", 
      deletedMaster: delMaster, 
      deletedBarangay: delBarangay 
    };

  } catch (err) {
    return { success: false, message: err.toString() };
  }
}

function updateRow(sheet, rowNumber, header, newObj) {
  var rowVals = header.map(function (col) {
    if (col === "Birthdate") {
      return formatDateText_(newObj[col]);
    }
    
    // FIXED: Don't overwrite Submitted On during updates - preserve original
    if (col === "Submitted On") {
      // If newObj has Submitted On, use it (preserve original), otherwise use current date
      return newObj[col] || formatTimestampPH_(new Date());
    }
    
    // FIXED: Ensure Updated On uses proper format
    if (col === "Updated On") {
      // Always format the Updated On date properly
      return formatTimestampPH_(newObj[col] || new Date());
    }
    
    // For Submitted By, Updated By, etc., use the value from newObj
    return newObj[col] || "";
  });

  sheet.getRange(rowNumber, 1, 1, header.length).setValues([rowVals]);
}

        }
      }
    }

    return null;
  }

/* ---------------------- Enhanced getRecords with filtering ---------------------- */
function getRecords(params) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // Pagination parameters
    var page = parseInt(params.page) || 1;
    var limit = parseInt(params.limit) || 500;
    var filterBarangay = (params.filterBarangay || "").trim();
    var searchQuery = (params.q || "").trim().toLowerCase();
    
    // Determine which sheet to use
    var sheet;
    var header;
    var totalRows;
    
    if (filterBarangay) {
      // User filtered by barangay - use barangay sheet
      sheet = ss.getSheetByName(filterBarangay);
      if (!sheet) {
        return { success: false, message: "Barangay not found: " + filterBarangay };
      }
      
      // Get header from sheet
      header = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      totalRows = sheet.getLastRow() - 1;
      
    } else {
      // No filter - use Master sheet
      sheet = ss.getSheetByName("Master");
      if (!sheet) {
        return { success: false, message: "Master sheet not found" };
      }
      
      header = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      totalRows = sheet.getLastRow() - 1;
    }
    
    // Calculate pagination
    var totalPages = Math.ceil(totalRows / limit);
    var startRow = 2 + ((page - 1) * limit); // +2 because row 1 is header
    
    // Ensure we don't go beyond available rows
    if (startRow > sheet.getLastRow()) {
      return {
        success: true,
        header: header,
        rows: [],
        pagination: {
          page: page,
          limit: limit,
          total: totalRows,
          pages: totalPages,
          hasNext: page < totalPages,
          hasPrev: page > 1
        }
      };
    }
    
    var endRow = Math.min(startRow + limit - 1, sheet.getLastRow());
    
    console.log("Loading rows " + startRow + " to " + endRow + " from " + 
                (filterBarangay || "Master") + " sheet");
    
    // Get ONLY the rows needed for this page
    var range = sheet.getRange(startRow, 1, endRow - startRow + 1, header.length);
    var values = range.getValues();
    
    // Apply search filter if needed (client-side would be better, but we'll do server-side)
    var rows = [];
    
    if (searchQuery) {
      // Find column indexes for search
      var headNameIdx = header.indexOf("Name of Household Head");
      var memberNameIdx = header.indexOf("Name of Household Member/s");
      var barangayIdx = header.indexOf("Barangay");
      var dataIdIdx = header.indexOf("Data ID");
      
      // Filter the current page
      for (var i = 0; i < values.length; i++) {
        var row = values[i];
        var headName = String(row[headNameIdx] || "").toLowerCase();
        var memberName = String(row[memberNameIdx] || "").toLowerCase();
        var barangay = String(row[barangayIdx] || "").toLowerCase();
        var dataId = String(row[dataIdIdx] || "");
        
        if (headName.includes(searchQuery) || 
            memberName.includes(searchQuery) || 
            barangay.includes(searchQuery) ||
            dataId.includes(searchQuery)) {
          
          var obj = {};
          for (var j = 0; j < header.length; j++) {
            if (header[j] === "Birthdate" || header[j].endsWith("_Date")) {
              obj[header[j]] = formatDateText_(row[j]);
            } else if (header[j] === "Submitted On") {
              obj[header[j]] = formatTimestampPH_(row[j]);
            } else {
              obj[header[j]] = row[j];
            }
          }
          rows.push(obj);
        }
      }
    } else {
      // No search filter - just format all rows in this page
      rows = values.map(function(r) {
        var obj = {};
        for (var i = 0; i < header.length; i++) {
          if (header[i] === "Birthdate" || header[i].endsWith("_Date")) {
            obj[header[i]] = formatDateText_(r[i]);
          } else if (header[i] === "Submitted On") {
            obj[header[i]] = formatTimestampPH_(r[i]);
          } else {
            obj[header[i]] = r[i];
          }
        }
        return obj;
      });
    }
    
    return {
      success: true,
      header: header,
      rows: rows,
      pagination: {
        page: page,
        limit: limit,
        total: totalRows,
        pages: totalPages,
        hasNext: page < totalPages,
        hasPrev: page > 1,
        currentSheet: filterBarangay || "Master",
        sheetType: filterBarangay ? "barangay" : "master"
      },
      filter: {
        barangay: filterBarangay || "all",
        search: searchQuery || ""
      }
    };
    
  } catch (err) {
    console.error("getRecords error:", err.toString());
    return { success: false, message: err.toString() };
  }
}

// Main virtual scrolling endpoint - FIXED to include member data
function getHouseholdsVirtual(params) {
  const startTime = new Date().getTime();

    var offset = parseInt(params.offset) || 0;
    var limit = parseInt(params.limit) || 100;
    var filterBarangay = (params.filterBarangay || "").trim();
    var searchQuery = (params.q || "").trim().toLowerCase();
    
    // Create cache key

    const cache = CacheService.getScriptCache();
    const cached = cache.get(cacheKey);
    
    if (cached) {
      const result = JSON.parse(cached);
      console.log(`⚡ Cache hit! ${offset}-${offset+limit}`);
      return result;
    }
    
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var master = ss.getSheetByName("Master");
    if (!master) return { success: false, message: "Master sheet not found" };
    
    // Get header
    var header = master.getRange(1, 1, 1, master.getLastColumn()).getValues()[0];
    
    // Column indexes
    var colIndex = {};
    for (var i = 0; i < header.length; i++) {
      colIndex[header[i]] = i;
    }
    
    // Get ALL data
    var allData = master.getRange(2, 1, master.getLastRow() - 1, header.length).getValues();
    
    // Group into households
    var households = {};
    for (var i = 0; i < allData.length; i++) {
      var row = allData[i];
      var dataID = String(row[colIndex["Data ID"]] || "");
      if (!dataID) continue;
      
      if (!households[dataID]) {
        households[dataID] = { 
          dataID: dataID, 
          head: null, 
          members: [] 
        };
      }
      
      var memberName = row[colIndex["Name of Household Member/s"]];
      var isHead = !memberName || String(memberName).trim() === "";
      
      if (isHead && !households[dataID].head) {
        households[dataID].head = row;
      } else if (!isHead) {
        households[dataID].members.push(row);
      }
    }
    
    // Convert to array
    var householdsArray = [];
    for (var id in households) {
      if (households[id].head) {
        householdsArray.push(households[id]);
      }
    }
    
    // Apply filters
    if (filterBarangay) {
      householdsArray = householdsArray.filter(function(h) {
        return String(h.head[colIndex["Barangay"]] || "").trim() === filterBarangay;
      });
    }
    
    if (searchQuery) {
      householdsArray = householdsArray.filter(function(h) {
        var headName = String(h.head[colIndex["Name of Household Head"]] || "").toLowerCase();
        var barangay = String(h.head[colIndex["Barangay"]] || "").toLowerCase();
        var dataId = String(h.dataID).toLowerCase();
        
        if (headName.includes(searchQuery) || barangay.includes(searchQuery) || dataId.includes(searchQuery)) {
          return true;
        }
        
        for (var j = 0; j < h.members.length; j++) {
          var memberName = String(h.members[j][colIndex["Name of Household Member/s"]] || "").toLowerCase();
          if (memberName.includes(searchQuery)) return true;
        }
        return false;
      });
    }
    
    // Sort by Data ID
    householdsArray.sort(function(a, b) { 
      return Number(a.dataID) - Number(b.dataID); 
    });
    
    var totalHouseholds = householdsArray.length;
    var hasMore = (offset + limit) < totalHouseholds;
    var pageHouseholds = householdsArray.slice(offset, offset + limit);
    
    // Format for response - NOW INCLUDING FULL MEMBER DATA
    var formattedHouseholds = [];
    for (var i = 0; i < pageHouseholds.length; i++) {
      var h = pageHouseholds[i];
      var head = {};
      
      // Get ALL head fields (preserve all columns)
      for (var key in colIndex) {
        var val = h.head[colIndex[key]];
        if (val !== undefined && val !== null && val !== "") {
          head[key] = val;
        }
      }
      
      // Get ALL member data (full objects)
      var members = [];
      for (var j = 0; j < h.members.length; j++) {
        var member = {};
        for (var key in colIndex) {
          var val = h.members[j][colIndex[key]];
          if (val !== undefined && val !== null && val !== "") {
            member[key] = val;
          }
        }
        members.push(member);
      }

/* ---------------------- PAGINATE BY COMPLETE HOUSEHOLDS ---------------------- */
function getHouseholdsByHeadPaginated(params) {
  const startTime = new Date().getTime();
  
  try {
    var page = parseInt(params.page) || 1;
    var headsPerPage = parseInt(params.headsPerPage) || 15;
    var filterBarangay = (params.filterBarangay || "").trim();
    var searchQuery = (params.q || "").trim().toLowerCase();
    
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var master = ss.getSheetByName("Master");
    if (!master) {
      return { success: false, message: "Master sheet not found" };
    }
    
    var fullHeader = master.getRange(1, 1, 1, master.getLastColumn()).getValues()[0];
    
    var colIndex = {};
    for (var i = 0; i < fullHeader.length; i++) {
      colIndex[fullHeader[i]] = i;
    }
    
    const dataIDIdx = colIndex["Data ID"];
    const headNameIdx = colIndex["Name of Household Head"];
    const memberCountIdx = colIndex["No. of Household Member/s"];
    const memberNameIdx = colIndex["Name of Household Member/s"];
    const barangayIdx = colIndex["Barangay"];
    
    var allData = master.getRange(2, 1, master.getLastRow() - 1, fullHeader.length).getValues();
    console.log("Total data rows read:", allData.length);
    
    // CRITICAL FIX: Group by Data ID, not by sequential order
    var households = {};
    
    for (var i = 0; i < allData.length; i++) {
      var row = allData[i];
      var dataID = String(row[dataIDIdx] || "").trim();
      if (!dataID) {
        console.log(`Row ${i + 2} has no Data ID, skipping`);
        continue;
      }
      
      if (!households[dataID]) {
        households[dataID] = { 
          dataID: dataID, 
          head: null, 
          members: [],
          headRowIndex: -1
        };
      }
      
      var headName = String(row[headNameIdx] || "").trim();
      var memberCount = String(row[memberCountIdx] || "").trim();
      var memberName = String(row[memberNameIdx] || "").trim();
      
      // Determine if this is a HEAD row
      // A row is a HEAD if:
      // 1. It has a member count (number) in the member count column, OR
      // 2. It's the first row for this Data ID, OR
      // 3. The head name is not empty and no head has been set yet
      
      var hasMemberCount = memberCount !== "" && !isNaN(parseInt(memberCount));
      var isHeadRow = false;
      
      if (!households[dataID].head) {
        // No head assigned yet for this Data ID
        if (hasMemberCount) {
          // Has member count - definitely a head
          isHeadRow = true;
        } else if (headName !== "" && memberName === "") {
          // Has head name but no member name - likely a head
          isHeadRow = true;
        } else if (headName !== "" && memberName !== "" && headName === memberName) {
          // Self-reference case
          isHeadRow = true;
        }
      }
      
      if (isHeadRow && !households[dataID].head) {
        // This is the head row
        households[dataID].head = row;
        households[dataID].headRowIndex = i;
        console.log(`✅ HEAD for Data ID ${dataID}: ${headName} (Member Count: ${memberCount || '0'})`);
      } else if (households[dataID].head && memberName !== "") {
        // This is a member row (has member name)
        households[dataID].members.push(row);
        console.log(`👤 MEMBER for Data ID ${dataID}: ${memberName}`);
      } else if (households[dataID].head && headName !== "" && memberName === "" && !hasMemberCount) {
        // This might be a duplicate head row - treat as member if it has no member count
        console.log(`⚠️ Duplicate head row for ${dataID} treated as member: ${headName}`);
      }
    }
    
    // Convert to array and ensure member counts are correct
    var householdsArray = [];
    for (var id in households) {
      var h = households[id];
      if (h.head !== null) {
        // Update the member count on the head row to match actual members
        if (memberCountIdx !== undefined && h.head[memberCountIdx] != h.members.length) {
          console.log(`📊 Updating member count for Data ID ${id}: ${h.head[memberCountIdx]} -> ${h.members.length}`);
          h.head[memberCountIdx] = h.members.length;
        }
        householdsArray.push(h);
      }
    }
    
    // Apply barangay filter
    if (filterBarangay && filterBarangay !== "") {
      householdsArray = householdsArray.filter(function(h) {
        return String(h.head[barangayIdx] || "").trim() === filterBarangay;
      });
    }
    
    // Apply search filter
    if (searchQuery) {
      householdsArray = householdsArray.filter(function(h) {
        var headName = String(h.head[headNameIdx] || "").toLowerCase();
        var barangay = String(h.head[barangayIdx] || "").toLowerCase();
        var dataId = String(h.dataID).toLowerCase();
        
        if (headName.includes(searchQuery) || barangay.includes(searchQuery) || dataId.includes(searchQuery)) {
          return true;
        }
        
        for (var j = 0; j < h.members.length; j++) {
          var memberName = String(h.members[j][memberNameIdx] || "").toLowerCase();
          if (memberName.includes(searchQuery)) return true;
        }
        return false;
      });
    }
    
    // Sort by Data ID
    householdsArray.sort(function(a, b) { 
      return Number(a.dataID) - Number(b.dataID); 
    });
    
    // Paginate
    var totalHouseholds = householdsArray.length;
    var totalPages = Math.ceil(totalHouseholds / headsPerPage);
    var startIndex = (page - 1) * headsPerPage;
    var endIndex = Math.min(startIndex + headsPerPage, totalHouseholds);
    var pageHouseholds = householdsArray.slice(startIndex, endIndex);
    
    // Build result rows
    var resultRows = [];
    for (var i = 0; i < pageHouseholds.length; i++) {
      var h = pageHouseholds[i];
      if (h.head) {
        var headRow = {};
        for (var j = 0; j < fullHeader.length; j++) {
          var colName = fullHeader[j];
          var value = h.head[j];
          if (value !== undefined && value !== null) {
            headRow[colName] = value;
          }
        }
        // Ensure member count is correct
        if (memberCountIdx !== undefined) {
          headRow[fullHeader[memberCountIdx]] = h.members.length;
        }
        resultRows.push(headRow);
      }
      for (var j = 0; j < h.members.length; j++) {
        var memberRow = {};
        for (var k = 0; k < fullHeader.length; k++) {
          var colName = fullHeader[k];
          var value = h.members[j][k];
          if (value !== undefined && value !== null) {
            // Clear member count for member rows
            if (colName === "No. of Household Member/s") {
              memberRow[colName] = "";
            } else {
              memberRow[colName] = value;
            }
          }
        }
        resultRows.push(memberRow);
      }
    }
    
    console.log(`✅ FINAL: ${totalHouseholds} households, ${resultRows.length} total rows`);
    console.log(`📊 Member counts: ${householdsArray.map(h => h.members.length).join(', ')}`);
    
    return {
      success: true,
      header: fullHeader,
      rows: resultRows,
      pagination: {
        page: page,
        headsPerPage: headsPerPage,
        totalHouseholds: totalHouseholds,
        totalPages: totalPages,
        currentHouseholds: pageHouseholds.length,
        currentRows: resultRows.length,
        hasNext: page < totalPages,
        hasPrev: page > 1
      },
      timestamp: Date.now()
    };
    
  } catch (err) {
    console.error("Error:", err);
    return { success: false, message: err.toString() };
  }
}

    supportData.labels.forEach(supportType => {
      supportIndexes[supportType] = headerMap[supportType];

      { name: "Dog", qty: "Dog Qty." },
      { name: "Cat", qty: "Cat Qty." },

      { name: "Cow", qty: "Cow Qty." },
      { name: "Carabao", qty: "Carabao Qty." },

      animalIndexes[animal.name] = headerMap[animal.name];
      if (animal.qty) {
        animalIndexes[animal.qty] = headerMap[animal.qty];
      }
    });
    
    console.log("\n=== PROCESSING ROWS ===");
    
    // Process each row
    for (let i = 0; i < rows.length; i++) {
      const row = rows[i];
      
      // Get basic values
      const dataID = String(row[dataIDIndex] || "").trim();
      const headName = String(row[headNameIndex] || "").trim();
      const memberName = String(row[memberNameIndex] || "").trim();
      const barangay = String(row[barangayIndex] || "Unknown").trim();

      const isHead = !memberName || memberName === "";
      
      // Update totals
      if (isHead) {
        totalHeads++;
      } else {
        totalMembers++;
      }
      
      // Track household

          heads: 0,
          members: 0,
          barangay: barangay
        };

        households[dataID].heads++;
      } else {
        households[dataID].members++;

          members: 0,
          total: 0

        barangayData[barangay].members++;

                              coord !== "undefined" && !coord.toLowerCase().includes("no") &&
                              (coord.match(/[-]?\d+\.?\d*[,\s]*[-]?\d+\.?\d*/) || coord.includes(',') || coord.includes('.'));
        if (hasValidCoord) {
          geographicData.withCoordinates++;

          geographicData.withoutCoordinates++;

        geographicData.withoutCoordinates++;
      }

        if (structuralValue === "yes" || structuralValue === "y" || structuralValue === "true") {
          calamityImpact.structuralDamage++;
        }
      }
      
      if (householdInjuryIndex !== undefined && row[householdInjuryIndex] !== undefined) {
        const injuryValue = String(row[householdInjuryIndex] || "").toLowerCase().trim();
        if (injuryValue === "yes" || injuryValue === "y" || injuryValue === "true") {
          calamityImpact.householdInjury++;
        }
      }
      
      if (habitabilityIndex !== undefined && row[habitabilityIndex] !== undefined) {
        const habitabilityValue = String(row[habitabilityIndex] || "").toLowerCase().trim();
        if (habitabilityValue === "yes" || habitabilityValue === "y" || habitabilityValue === "true") {
          calamityImpact.habitability++;
        }
      }
      
      if (lossOfLivelihoodIndex !== undefined && row[lossOfLivelihoodIndex] !== undefined) {
        const livelihoodValue = String(row[lossOfLivelihoodIndex] || "").toLowerCase().trim();
        if (livelihoodValue === "yes" || livelihoodValue === "y" || livelihoodValue === "true") {
          calamityImpact.lossOfLivelihood++;
        }
      }
      
      // Calamity data by barangay for charts
      if (structuralDamageIndex !== undefined && row[structuralDamageIndex] !== undefined) {
        const damageValue = String(row[structuralDamageIndex] || "").toLowerCase().trim();
        const isDamaged = damageValue === "yes" || damageValue === "y" || damageValue === "true";
        
        if (!structuralDamageByBarangay[barangay]) {
          structuralDamageByBarangay[barangay] = { yes: 0, no: 0 };
        }
        if (isDamaged) {
          structuralDamageByBarangay[barangay].yes++;
        } else {
          structuralDamageByBarangay[barangay].no++;
        }
      }
      
      if (householdInjuryIndex !== undefined && row[householdInjuryIndex] !== undefined) {
        const injuryValue = String(row[householdInjuryIndex] || "").toLowerCase().trim();
        const hasInjury = injuryValue === "yes" || injuryValue === "y" || injuryValue === "true";

        if (hasInjury) {
          householdInjuryByBarangay[barangay].yes++;
        } else {
          householdInjuryByBarangay[barangay].no++;

      }

        const hasHabitabilityIssue = habitabilityValue === "yes" || habitabilityValue === "y" || habitabilityValue === "true";
        
        if (!habitabilityByBarangay[barangay]) {
          habitabilityByBarangay[barangay] = { yes: 0, no: 0 };
        }
        if (hasHabitabilityIssue) {
          habitabilityByBarangay[barangay].yes++;
        } else {
          habitabilityByBarangay[barangay].no++;

          lossLivelihoodByBarangay[barangay] = { yes: 0, no: 0 };

        }

      let ageYears = null;

          ageYears = Number(ageVal);

      if ((ageYears === null || ageYears === undefined) && birthdateIndex !== undefined) {

            } else {

            }
            disabilityTypeTotals[disabilityTypeValue] = (disabilityTypeTotals[disabilityTypeValue] || 0) + 1;
          }

if (statusIndex !== undefined && statusIndex !== -1) {

        if (!comparisonData[barangay]) {
            comparisonData[barangay] = {};
        }
        comparisonData[barangay][statusValue] = (comparisonData[barangay][statusValue] || 0) + 1;
    }
}

      if (registrationIndex !== undefined && row[registrationIndex] !== undefined) {
        const regValue = String(row[registrationIndex] || "").toLowerCase().trim();
        let regType = 3; // Default: Non Registered
        
        if (regValue.includes("fishr") && regValue.includes("rsbsa")) {
          regType = 2; // Both Registered
        } else if (regValue.includes("fishr")) {
          regType = 0; // FishR Registered
        } else if (regValue.includes("rsbsa")) {
          regType = 1; // RSBSA Registered
        }
        
        if (isHead) {
          registrationData.heads[regType]++;
        } else {
          registrationData.members[regType]++;
        }
      }
      
      // SUPPORT DATA
      supportData.labels.forEach((supportType, index) => {
        const supportIdx = supportIndexes[supportType];
        if (supportIdx !== undefined && row[supportIdx] !== undefined) {
          const value = String(row[supportIdx] || "").toLowerCase().trim();
          let hasSupport = false;
          
          if (supportType === 'AICS') {
            hasSupport = value !== "";
          } else {
            hasSupport = value === "yes" || value === "y" || value === "true";
          }
          
          if (hasSupport) {
            if (isHead) {
              supportData.heads[index]++;
            } else {
              supportData.members[index]++;
            }
          }
        }
      });
      
      // ADDITIONAL INFORMATION DATA
      if (soloParentingIndex !== undefined && row[soloParentingIndex] !== undefined) {
        const soloValue = String(row[soloParentingIndex] || "").toLowerCase().trim();
        if (soloValue === "yes" || soloValue === "y" || soloValue === "true") {
          if (isHead) additionalData.heads[0]++;
          else additionalData.members[0]++;
          soloParentingTotal++;
        }
      }
      
      if (lactatingIndex !== undefined && row[lactatingIndex] !== undefined) {
        const lactatingValue = String(row[lactatingIndex] || "").toLowerCase().trim();
        if (lactatingValue === "yes" || lactatingValue === "y" || lactatingValue === "true") {
          if (isHead) additionalData.heads[1]++;
          else additionalData.members[1]++;
          lactatingTotal++;
        }
      }
      
      if (pregnantIndex !== undefined && row[pregnantIndex] !== undefined) {
        const pregnantValue = String(row[pregnantIndex] || "").toLowerCase().trim();
        if (pregnantValue === "yes" || pregnantValue === "y" || pregnantValue === "true") {
          if (isHead) additionalData.heads[2]++;
          else additionalData.members[2]++;
          pregnantTotal++;
        }

      if (insideAssemblyIndex !== undefined && row[insideAssemblyIndex] !== undefined) {
        const insideValue = String(row[insideAssemblyIndex] || "").toLowerCase().trim();
        if (insideValue === "yes" || insideValue === "y" || insideValue === "true") {
          if (isHead) assemblyData.heads[0]++;
          else assemblyData.members[0]++;
        }
      }
      
      if (outsideAssemblyIndex !== undefined && row[outsideAssemblyIndex] !== undefined) {
        const outsideValue = String(row[outsideAssemblyIndex] || "").toLowerCase().trim();
        if (outsideValue === "yes" || outsideValue === "y" || outsideValue === "true") {
          if (isHead) assemblyData.heads[1]++;
          else assemblyData.members[1]++;
        }
      }
      
      // TRANSPORTATION DATA
      if (taalVolcanoIndex !== undefined && row[taalVolcanoIndex] !== undefined) {
        const taalValue = String(row[taalVolcanoIndex] || "").toLowerCase().trim();
        if (taalValue === "yes" || taalValue === "y" || taalValue === "true") {
          if (isHead) transportData.heads[0]++;
          else transportData.members[0]++;
        }
      }
      
      if (typhoonIndex !== undefined && row[typhoonIndex] !== undefined) {
        const typhoonValue = String(row[typhoonIndex] || "").toLowerCase().trim();
        if (typhoonValue === "yes" || typhoonValue === "y" || typhoonValue === "true") {
          if (isHead) transportData.heads[1]++;
          else transportData.members[1]++;
        }
      }
      
      if (floodIndex !== undefined && row[floodIndex] !== undefined) {
        const floodValue = String(row[floodIndex] || "").toLowerCase().trim();
        if (floodValue === "yes" || floodValue === "y" || floodValue === "true") {
          if (isHead) transportData.heads[2]++;
          else transportData.members[2]++;

      if (landslideIndex !== undefined && row[landslideIndex] !== undefined) {

        } else if (ofwValue !== "") {
          ofwData["Not Specified"]++;
        }

      // EDUCATION DATA
      if (educationIndex !== undefined && row[educationIndex] !== undefined) {
        const eduValue = String(row[educationIndex] || "").trim();
        if (eduValue) {
          educationData[eduValue] = (educationData[eduValue] || 0) + 1;
        }
      }
      
      // LIVESTOCK DATA (only for heads)
      if (isHead) {
        animalTypes.forEach(animal => {
          const animalIdx = animalIndexes[animal.name];
          if (animalIdx !== undefined && row[animalIdx] !== undefined) {
            const hasAnimal = String(row[animalIdx] || "").toLowerCase().trim();
            if (hasAnimal === "yes" || hasAnimal === "y" || hasAnimal === "true") {

                if (row[qtyIdx] !== undefined) {

                      quantity = qtyNum;

              livestockData[animal.name] = (livestockData[animal.name] || 0) + quantity;

    console.log("\n=== POST-PROCESSING DATA ===");
    
    // Process disability type data
    const disabilityTypeData = {
      labels: [],
      heads: [],
      members: []
    };
    
    const allDisabilityTypes = new Set([
      ...Object.keys(disabilityTypeTemp.heads),

      labels: [],

    Object.entries(disabilityTypeTotals).sort((a, b) => b[1] - a[1]).forEach(([type, count]) => {

      disabilityTypeTotalsData.values.push(count);
    });
    
    // Prepare age group data
    const validAgeGroups = [];
    const validAgeValues = [];
    const validAgeHeads = [];
    const validAgeMembers = [];
    
    Object.entries(ageGroups).forEach(([group, data]) => {
      if (data.total > 0) {
        validAgeGroups.push(group);
        validAgeValues.push(data.total);
        validAgeHeads.push(data.heads);
        validAgeMembers.push(data.members);
      }
    });
    
    // Prepare barangay data
    const barangayLabels = Object.keys(barangayData);
    const barangayHeads = barangayLabels.map(b => barangayData[b].heads);
    const barangayMembers = barangayLabels.map(b => barangayData[b].members);
    const barangayTotals = barangayLabels.map(b => barangayData[b].total);

    const lossLivelihoodChartData = prepareCalamityBarangayData(lossLivelihoodByBarangay);
    
const stageChartData = objectToChartData(stageData);
const statusHeadsChartData = {
    labels: Object.keys(statusData.heads),
    values: Object.values(statusData.heads)

const statusMembersChartData = {
    labels: Object.keys(statusData.members),
    values: Object.values(statusData.members)
};

// Debug log
console.log("Status Heads Data:", statusHeadsChartData);
console.log("Status Members Data:", statusMembersChartData);
    const livestockChartData = objectToChartData(livestockData, 10);
    const educationChartData = objectToChartData(educationData);
    const employmentChartData = objectToChartData(employmentData);
    
    // Use status comparison data (categorized into Common Sickness and Severe Illness)
    const comparisonChartData = prepareIllnessComparisonData(comparisonData, 10);
    
    // Create aggregated response

      barangay: {
        labels: barangayLabels,
        heads: barangayHeads,
        members: barangayMembers,
        totals: barangayTotals
      },
      totals: {

        members: totalMembers,
        total: totalHeads + totalMembers
      },

        labels: validAgeGroups,
        values: validAgeValues,
        heads: validAgeHeads,
        members: validAgeMembers
      },
      stage: stageChartData,
      pwd: pwdData,
      status: {
        heads: statusHeadsChartData,
        members: statusMembersChartData
      },
      support: supportData,
      registration: registrationData,
      additional: additionalData,
      assembly: assemblyData,
      transport: transportData,
      disabilityType: disabilityTypeData,
      disabilityTypeTotals: disabilityTypeTotalsData,
      livestock: livestockChartData,
      ofw: ofwData,
      education: educationChartData,
      employment: employmentChartData,
      comparison: comparisonChartData,
      geographic: geographicData,
      structuralDamage: structuralDamageChartData,
      householdInjury: householdInjuryChartData,
      habitability: habitabilityChartData,
      lossLivelihood: lossLivelihoodChartData,
      calamityImpact: calamityImpact
    };
    
    const processingTime = Date.now() - startTime;
    console.log(`\n=== FINISHED in ${processingTime}ms ===`);
    console.log("Final totals - Heads:", totalHeads, "Members:", totalMembers, "Total:", totalHeads + totalMembers);
    console.log("✅ Employment Data:", employmentData);
    console.log("✅ Registration Data:", registrationData);
    console.log("✅ Status Data - Heads:", Object.keys(statusData.heads).length, "categories");
    console.log("✅ Status Data - Members:", Object.keys(statusData.members).length, "categories");
    
    return {
      success: true,
      aggregated: aggregated,
      recordCount: rows.length,
      householdCount: Object.keys(households).length,
      processingTime: processingTime,
      timestamp: new Date().toISOString(),
      debugInfo: {
        totalRows: rows.length,
        totalHeads: totalHeads,
        totalMembers: totalMembers,
        barangayCount: barangayLabels.length,
        disabilityTypes: disabilityTypeData.labels.length,
        additionalInfo: {
          soloParenting: additionalData.heads[0] + additionalData.members[0],
          lactating: additionalData.heads[1] + additionalData.members[1],
          pregnant: additionalData.heads[2] + additionalData.members[2]
        },
        assemblyInfo: {
          inside: assemblyData.heads[0] + assemblyData.members[0],
          outside: assemblyData.heads[1] + assemblyData.members[1]
        },
        transportInfo: {
          taalVolcano: transportData.heads[0] + transportData.members[0],
          typhoon: transportData.heads[1] + transportData.members[1],
          flood: transportData.heads[2] + transportData.members[2],
          landslide: transportData.heads[3] + transportData.members[3],
          earthquake: transportData.heads[4] + transportData.members[4],
          fire: transportData.heads[5] + transportData.members[5]
        },
        geographicInfo: geographicData,
        calamityInfo: calamityImpact,
        employmentInfo: employmentData
      }
    };
    
  } catch (error) {
    console.error("Error in getAccurateChartData:", error.toString());
    console.error("Stack trace:", error.stack);
    
    return {
      success: false,
      error: error.toString(),
      message: "Failed to get chart data",
      timestamp: new Date().toISOString()
    };
  }
}

// ============ HELPER FUNCTIONS ============

function prepareCalamityBarangayData(calamityByBarangay) {
  const barangaysWithData = Object.keys(calamityByBarangay);
  
  if (barangaysWithData.length === 0) {
    return { labels: [], yes: [], no: [] };
  }
  
  const labels = [];
  const yesData = [];
  const noData = [];
  
  barangaysWithData.forEach(barangay => {
    labels.push(barangay);
    yesData.push(calamityByBarangay[barangay].yes || 0);
    noData.push(calamityByBarangay[barangay].no || 0);
  });
  
  return {
    labels: labels,
    yes: yesData,
    no: noData
  };
}

function prepareIllnessComparisonData(comparisonObj, limitBarangays = 10) {
  // Define which status values belong to Common Sickness vs Severe Illness
  const commonSicknessValues = [
    "Cough and colds", "Fever", "Headache", "Diarrhea", 
    "Stomach ache / Hyperacidity", "Skin disease (rashes, fungal infection)",
    "Urinary Tract Infection (UTI)"

function calculateAgeFromRow(rowObj) {
  const ageVal = rowObj["Age"];
  const birthRaw = rowObj["Birthdate"] || rowObj["Birth Date"];
  
  let age = null;
  
  // Try to get age from Age field
  if (ageVal !== "" && ageVal !== null && ageVal !== undefined && !isNaN(Number(ageVal))) {
    age = Number(ageVal);
  }
  
  // If no age, try to calculate from birthdate
  if ((age === null || age === undefined) && birthRaw) {
    try {
      const birthDate = new Date(birthRaw);
      if (!isNaN(birthDate.getTime())) {
        const now = new Date();
        age = now.getFullYear() - birthDate.getFullYear();
        // Adjust if birthday hasn't occurred this year
        if (now.getMonth() < birthDate.getMonth() || 
            (now.getMonth() === birthDate.getMonth() && now.getDate() < birthDate.getDate())) {
          age--;
        }
      }
    } catch (e) {
      console.log("Could not parse birthdate:", birthRaw);
    }
  }
  
  // Determine age group
  if (age === null || age === undefined || age === "") {
    return { group: "Unknown", index: 4 };
  }
  if (age < 1) {
    return { group: "Infant (0-11 mos)", index: 0 };
  }
  if (age <= 17) {
    return { group: "School Children (≤17)", index: 1 };
  }
  if (age >= 60) {
    return { group: "Elderly (60+)", index: 3 };
  }
  return { group: "Adult (18-59)", index: 2 };
}

function processSupport(rowObj, supportData, isHead) {
  const supportKeys = ["TUPAD", "Local 4Ps", "National 4Ps", "Local Social Pension", 
                      "National Social Pension", "Solo Parent Assistance", "Cash Subsidy", 
                      "Relief Goods", "AICS"];
  
  supportKeys.forEach((key, index) => {
    const value = (rowObj[key] || "").toString().toLowerCase().trim();
    let hasSupport = false;
    
    if (key === 'AICS') {
      // For AICS, any value means yes
      hasSupport = value !== "";
    } else {
      hasSupport = value === "yes" || value === "y" || value === "true";
    }
    
    if (hasSupport) {
      if (isHead) {
        supportData.heads[index]++;
      } else {
        supportData.members[index]++;
      }
    }
  });
}

function processRegistration(rowObj, regData, isHead) {
  const regValue = (rowObj["FishR & RSBSA"] || "").toString().toLowerCase().trim();
  
  let regIndex = 3; // Default: Non Registered
  
  if (regValue.includes("fishr") && regValue.includes("rsbsa")) {
    regIndex = 2; // Both Registered
  } else if (regValue.includes("fishr")) {
    regIndex = 0; // FishR Registered
  } else if (regValue.includes("rsbsa")) {
    regIndex = 1; // RSBSA Registered
  }
  
  if (isHead) {
    regData.heads[regIndex]++;
  } else {
    regData.members[regIndex]++;
  }
}

function processLivestock(rowObj, livestockData) {
  const animals = ["Dog", "Cat", "Pig", "Cow", "Carabao", "Chicken"];
  
  animals.forEach(animal => {
    const hasAnimal = (rowObj[animal] || "").toString().toLowerCase().trim();
    if (hasAnimal === "yes" || hasAnimal === "y" || hasAnimal === "true") {
      livestockData[animal] = (livestockData[animal] || 0) + 1;

  const others = (rowObj["Others"] || "").toString().trim();
  if (others) {

  }
}

    .sort((a, b) => b[1] - a[1])
    .slice(0, limitBarangays)
    .map(entry => entry[0]);

  const allStatuses = new Set();
  for (const statusCounts of Object.values(comparisonObj)) {
    for (const status of Object.keys(statusCounts)) {
      allStatuses.add(status);
    }
  }
  
  const statusList = Array.from(allStatuses).sort();
  
  // Create dataset for each status
  const datasets = [];

      return comparisonObj[barangay] && comparisonObj[barangay][status]

        : 0;
    });
    
    datasets.push({
      label: status,
      data: data
    });
  }
  
  return {
    labels: topBarangays,
    datasets: datasets
  };
}

/* ---------------------- Helper: Parse any date format ---------------------- */
function parseAnyDate(val) {
  if (!val) return null;
  
  // If already a Date object
  if (val instanceof Date) return val;
  
  val = String(val).trim();
  if (!val) return null;
  
  // mm/dd/yyyy format
  if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(val)) {
    const parts = val.split('/');
    const month = parseInt(parts[0]) - 1; // Months are 0-indexed
    const day = parseInt(parts[1]);
    const year = parseInt(parts[2]);
    return new Date(year, month, day);
  }
  
  // yyyy-mm-dd format
  if (/^\d{4}-\d{1,2}-\d{1,2}/.test(val)) {
    return new Date(val);
  }
  
  // Try to parse as a date
  const d = new Date(val);
  return isNaN(d.getTime()) ? null : d;
}

/* ---------------------- HELPER: Delete all rows for a Data ID (FAST) ---------------------- */
function deleteAllRowsForDataID(sheet, dataID) {
  if (!sheet) return 0;
  
  const data = sheet.getDataRange().getValues();
  const rowsToDelete = [];
  
  // Find rows to delete (start from bottom)
  for (let i = data.length - 1; i >= 0; i--) {
    if (String(data[i][0]) === String(dataID)) {
      rowsToDelete.push(i + 1); // +1 because sheet rows are 1-indexed
    }
  }
  
  // Delete from bottom to top
  rowsToDelete.sort((a, b) => b - a).forEach(rowNum => {
    sheet.deleteRow(rowNum);
  });
  
  return rowsToDelete.length;
}

    const subfolder = getCalamitySubfolder(category);
    if (!subfolder) {
      console.error(`Cannot find subfolder for category: ${category}`);
      return null;
    }
    
    console.log(`Uploading to folder: ${subfolder.getName()}`);
    
    // Decode base64 file data
    let blob;
    if (fileData.indexOf('base64,') !== -1) {
      const base64Data = fileData.split('base64,')[1];
      blob = Utilities.newBlob(Utilities.base64Decode(base64Data));
    } else {
      blob = Utilities.newBlob(Utilities.base64Decode(fileData));
    }
    
    // Set file name with timestamp and dataID for easy lookup
    const timestamp = new Date().getTime();
    const safeFileName = `${dataID}_${category}_${timestamp}_${fileName.replace(/[^a-zA-Z0-9.-]/g, '_')}`;
    blob.setName(safeFileName);
    
    // Set MIME type
    const fileExt = fileName.split('.').pop().toLowerCase();
    const mimeTypes = {
      'jpg': 'image/jpeg', 'jpeg': 'image/jpeg', 'png': 'image/png',
      'gif': 'image/gif', 'bmp': 'image/bmp', 'tiff': 'image/tiff',
      'pdf': 'application/pdf'
    };
    blob.setContentType(mimeTypes[fileExt] || 'application/octet-stream');
    
    // Upload to the subfolder
    const file = subfolder.createFile(blob);
    
    // Set description for easy reference
    file.setDescription(`Data ID: ${dataID} | Category: ${category} | Uploaded: ${new Date().toISOString()}`);
    
    console.log(`✅ File uploaded: ${safeFileName} (ID: ${file.getId()})`);
    
    return file.getId();
    
  } catch (err) {
    console.error(`Error uploading ${category} file:`, err);
    return null;
  }
}

function loadAllUsersToPropertiesCache() {
  console.log("🔄 Loading all users to Properties cache...");
  const startTime = Date.now();
  
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const usersSheet = ss.getSheetByName("Users");
    if (!usersSheet) {
      console.error("Users sheet not found");
      return false;
    }
    
    const lastCol = Math.min(usersSheet.getLastColumn(), 10);
    const dataRange = usersSheet.getRange(1, 1, usersSheet.getLastRow(), lastCol);
    const values = dataRange.getValues();
    
    if (values.length < 2) return false;
    
    const headers = values[0];
    
    let usernameIdx = -1, passwordIdx = -1, fullNameIdx = -1, roleIdx = -1, barangayIdx = -1, mustChangeIdx = -1;
    
    for (let i = 0; i < headers.length; i++) {
      const header = String(headers[i]).trim().toLowerCase();
      if (header === "username") usernameIdx = i;
      else if (header === "passwordhash" || header === "password") passwordIdx = i;
      else if (header === "fullname" || header === "name") fullNameIdx = i;
      else if (header === "role") roleIdx = i;
      else if (header === "assignedbarangay" || header === "barangay") barangayIdx = i;
      else if (header === "mustchangepassword") mustChangeIdx = i;
    }
    
    let loadedCount = 0;
    const props = PropertiesService.getScriptProperties();
    
    for (let i = 1; i < values.length; i++) {
      const row = values[i];
      const username = String(row[usernameIdx] || "").trim();
      if (!username) continue;
      
      const mustChangePassword = (mustChangeIdx !== -1 && (row[mustChangeIdx] === true || row[mustChangeIdx] === "TRUE"));
      
      const userData = {
        passwordHash: String(row[passwordIdx] || "").trim(),
        fullName: (fullNameIdx !== -1 && row[fullNameIdx]) ? String(row[fullNameIdx]).trim() : username,
        role: (roleIdx !== -1 && row[roleIdx]) ? String(row[roleIdx]).trim().toLowerCase() : "user",
        assignedBarangay: (barangayIdx !== -1 && row[barangayIdx]) ? String(row[barangayIdx]).trim() : "",
        mustChangePassword: mustChangePassword
      };
      
      props.setProperty(`user_props_${username}`, JSON.stringify(userData));
      loadedCount++;
    }
    
    props.setProperty('users_cache_timestamp', Date.now().toString());
    props.setProperty('users_cache_version', '2');
    
    const elapsed = Date.now() - startTime;
    console.log(`✅ Loaded ${loadedCount} users to Properties cache in ${elapsed}ms`);
    return true;

function clearAllStaleSessions() {
  console.log("🔄 Clearing all stale sessions...");
  
  const props = PropertiesService.getScriptProperties();
  const allProps = props.getProperties();
  let sessionCount = 0;
  let clearedCount = 0;
  
  // Clear all session_* properties
  for (const key in allProps) {
    if (key.startsWith('session_')) {
      props.deleteProperty(key);
      sessionCount++;
      clearedCount++;
      console.log(`  Deleted: ${key}`);
    }
  }
  
  // Also clear pending force logout requests
  props.deleteProperty('pendingForceLogoutRequests');
  
  // Update UserActivity sheet to mark all as offline
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const activitySheet = ss.getSheetByName("UserActivity");
    if (activitySheet) {
      const lastRow = activitySheet.getLastRow();
      for (let i = 2; i <= lastRow; i++) {
        activitySheet.getRange(i, 6).setValue("Offline"); // IsOnline
        activitySheet.getRange(i, 10).setValue("Inactive"); // Status
      }
      console.log(`  Updated ${lastRow - 1} rows in UserActivity sheet`);
    }
  } catch (e) {
    console.log("  Could not update UserActivity sheet:", e.message);
  }
  
  console.log(`✅ Cleared ${clearedCount} stale sessions`);
  
  return { 
    success: true, 
    sessionsCleared: clearedCount,
    message: `Cleared ${clearedCount} stale sessions`
  };
}

// Run this via a time-based trigger (every hour)
function cleanupStaleSessions() {
  console.log("🔄 Running stale session cleanup...");
  
  const props = PropertiesService.getScriptProperties();
  const allProps = props.getProperties();
  let cleanedCount = 0;
  
  for (const key in allProps) {
    if (key.startsWith('session_')) {
      try {
        const sessionData = JSON.parse(allProps[key]);
        const lastActivity = new Date(sessionData.lastActivity);
        const inactiveHours = (Date.now() - lastActivity.getTime()) / (1000 * 60 * 60);
        
        // Clean up sessions inactive for more than 2 hours (browser likely closed)
        if (inactiveHours > 2) {
          const username = key.replace('session_', '');
          console.log(`🧹 Cleaning stale session for: ${username} (inactive for ${Math.floor(inactiveHours)} hours)`);
          invalidateSession(username);
          cleanedCount++;
        }
      } catch(e) {
        console.error("Error cleaning session:", key, e);
      }
    }
  }
  
  console.log(`✅ Cleaned up ${cleanedCount} stale sessions`);
  return { success: true, cleaned: cleanedCount };
}
